אָסוּד
אָסוּד הָיָה חֵיל מִשְׁמָר וִיחִידָה צְבָאִית שֶׁל חַיָּילִים מִמּוֹצָא אָלָאנִי. שֵׁם הַשֵּׁבֶט הַמּוֹנְגוֹלִי הוּא צוּרַת הָרַבִּים שֶׁל אָס, שְׁמָם שֶׁל הָאָלָאנִים בַּעֲרָבִית.
לְאַחַר הַפְּלִישָׁה הַמּוֹנְגוֹלִית לְרוּס (רוּס שֶׁל קִיֶּיב) נִכְנְעוּ רַבִּים מֵהָאָלָאנִים לָאִימְפֶּרְיָה הַמּוֹנְגוֹלִית. חֵלֶק מֵהֶם הִמְשִׁיךְ לְהִתְנַגֵּד לְאוּרְדַת הַזָּהָב. לוֹחֲמִים רַבִּים הוֹעׇבְרוּ מִצְּפוֹן הַקַּווְקָז לְמוֹנְגוֹלְיָה. הוֹעׇלְתָה גַּם הַשְׁעָרָה כִּי הֵם עָזְרוּ לַאֲדוֹנֵיהֶם הַחֲדָשִׁים בְּמִלְחַמְתָּם כְּנֶגֶד הַצֶּ'רְקֵסִים וְכִי הֵם הִשְׁתַּתְּפוּ בַּפְּלִישָׁה הַמּוֹנְגוֹלִית לְאֵירוֹפָּה. תַּחַת שִׁלְטוֹנוֹ שֶׁל מוֹנְגְקֶה חָאן הֵם נִלְחֲמוּ כְּנֶגֶד שׁוֹשֶׁלֶת סוֹנְג הַסִּינִית וּמַמְלֶכֶת נַנְזָאוֹ בְּיוּנָאן (1258-1259). הַמִּשְׁמָר הַקֵּיסָרִי הָאָלָאנִי הָיָה מְחוּלָּק לִשְׁנֵי יְחִידוֹת שֶׁהַמִּפְקָדָה שֶׁלָּהֶם הָיְיתָה בְּחַרְחוֹרִין.
לְאַחַר הַכְתָּרָתוֹ שֶׁל קוּבְּלַאי חָאן, הִשְׁתַּתְּפוּ הָאָלָאנִים בַּמַּעֲרָכָה כְּנֶגֶד אָרִיק בּוּקָה, אֲחִיהֶם שֶׁל מוֹנְגְקֶה חָאן וְקוּבְּלַאי חָאן וְהַטּוֹעֵן לַכֶּתֶר הַמּוֹנְגוֹלִי, וּמְאוּחָר יוֹתֵר הִשְׁתַּתְּפוּ בַּמַּעֲרָכָה כְּנֶגֶד קָאִידוֹ תַּחַת שׁוֹשֶׁלֶת יוּאָן. הֵם סָבְלוּ אֲבֵידוֹת כְּבֵדוֹת מֵהַהִתְנַגְּדוּת שֶׁל סוֹנְג כַּאֲשֶׁר שֵׁרְתוּ תַּחַת הַנְהָגָתָם שֶׁל בָּאיָאן מִבָּאָרִין[1] וְאָג'וּ.
בִּסְבִיבוֹת 1271 אִירְגֵּן אוֹתָם קוּבְּלַאי חָאן כְּ"מִשְׁמַר גָּאסִין" ("מִשְׁמָר אָלָאנִי") יַחַד עִם חֵלֶק מֵהַקִּיפְּצָקִים. הַמִּשְׁמָר הָאָלָאנִי הִגִּיעַ לְשִׂיאוֹ בִּתְקוּפַת שִׁלְטוֹנוֹ שֶׁל הַקֵּיסָר הַסִּינִי גִיאָטוּ חָאן. הֵם נוֹתְרוּ בַּעֲלֵי הַשְׁפָּעָה בְּמוֹנְגוֹלְיָה עַד לַמֵּאָה הַ-15, כַּאֲשֶׁר הָאוֹיְרֵטִים כָּבְשׁוּ אֶת מוֹנְגוֹלְיָה.
כַּיּוֹם חַיָּה קְבוּצָה קְטַנָּה שֶׁל שֵׁבֶט הָאָסוּד בְּמוֹנְגוֹלְיָה הַפְּנִימִית.
 אַדְרִיכָלוּת קְלָאסִית
אַדְרִיכָלוּת קְלָאסִית הוּא שֵׁם כּוֹלֵל לְסִגְנוֹנוֹת אַדְרִיכָלִיִּים שֶׁאִיפְיְינוּ אֶת אֵזוֹר הַיָּם הַתִּיכוֹן בָּעֵת הָעֲתִיקָה בְּמֶשֶׁךְ תְּקוּפָה שֶׁל כְּאֶלֶף שָׁנָה.
הָאַדְרִיכָלוּת הַקְלָאסִית מִתְיַיחֶסֶת לַסִּגְנוֹן הָאַדְרִיכָלִי אֲשֶׁר פּוּתַּח בְּיָוָון הָעַתִּיקָה כְּבָר בַּתְּקוּפָה הָאַרְכָאִית, הָאַדְרִיכָלוּת הַהֵלֵנִיסְטִית שֶׁאִיפְיְינָה אֶת מִזְרַח הַיָּם הַתִּיכוֹן וְאֶת אֵזוֹר הַמִּזְרָח הַתִּיכוֹן אֲשֶׁר נִכְבְּשׁוּ עַל יְדֵי אָלֶכְּסַנְדֶר מוֹקְדוֹן וְאֶת הָאַדְרִיכָלוּת הָרוֹמִית עַד לִנְפִילַת הָאִימְפֶּרְיָה הָרוֹמִית.
הָאַדְרִיכָלוּת הַקְּלָאסִית הִיוְּותָה בָּסִיס לְכׇל סִגְנוֹנוֹת הַבְּנִיָּיה בְּתַרְבּוּת הַמַּעֲרָב עַד הַמֵּאָה הָ-20. הַמִּבְנִים הַקְּלָאסִים עֲלֵיהֶם מְדוּבָּר הֵם בְּעִיקָּר מִקְדָּשִׁים וְכֵן גַּם מוֹסְדוֹת שִׁלְטוֹן וְתֵיאַטְרָאוֹת אֲשֶׁר לֹא הָיוּ קַיָּימִים עַד אוֹתָהּ הָעֵת.
הָאַדְרִיכָלוּת בַּתְּקוּפָה הַקְּלָאסִית שֶׁל יָוָון הָעַתִּיקָה (הַמֵּאָה הַ-5 לפנה"ס) כְּבָר הָיְיתָה בְּרוּרָה וּמְזוּקֶּקֶת. הַסִּגְנוֹן הִגְדִּיר כְּלָלִים מְאוֹד בְּרוּרִים לִבְנִיָּיה אֲשֶׁר הִתְבַּסְּסוּ עַל סֵדֶר וְסִימֶטְרִיָּה. נוּסְּחוּ סְדָרִים אֲשֶׁר הִגְדִּירוּ אֶת הַפְּרוֹפּוֹרְצִיּוֹת הַמְּדוּיָּקוֹת וְאֶת הַתְאָמַת הָאוֹרְנוּמַנְטִיקָה (קִישּׁוּטִים) לַמִּבְנֶה.
בַּמֵּאָה הָ-4 לפנה"ס כָּבַשׁ אָלֶכְּסַנְדֶר מוֹקְדוֹן שְׁטָחִים עֲצוּמִים מֵהַבַּלְקָן וְעַד הוֹדּוּ. בְּמִסְגֶּרֶת הֲפָצַת הַתַּרְבּוּת הַהֵלֵנִיסְטִית נִבְנוּ עָרִים וּמִבְנִים בַּסִּגְנוֹן הַיְּוָונִי הַקְּלָאסִי. שְׂרִידִים אַרְכֵאוֹלוֹגִיִּים שֶׁל תֵּיאַטְרָאוֹת נִמְצָאִים בְּכׇל רַחֲבֵי הַמִּזְרָח הַתִּיכוֹן וְעַד לְאַפְגָּנִיסְטָן.
הָאַדְרִיכָלוּת הָרוֹמִית אִימְּצָה אֶת הַסִּגְנוֹן וְהִמְשִׁיכָה לְשַׁכְלְלוֹ וּלְפַתְּחוֹ עַד הַמֵּאָה הַ-5 לַסְּפִירָה.
 אִיץ וּמִיץ
אִיץ וּמִיץ הָיְיתָה תּוֹכְנִית טֵלֵוִויזְיָה לִילָדִים וָנוֹעַר, שֶׁשּׁוּדְּרָה בַּשָּׁנִים 1996–1997 בַּעֲרוּץ 2.
הַתּוֹכְנִית נָסַובָּה סְבִיב שְׁנֵי חַיְיזָרִים בְּשֵׁם אִיץ וּמִיץ שֶׁבָּאוּ מִכּוֹכָב בְּשֵׁם "הָפוּכוּלוּס" וְנָחֲתוּ עַל גַּג שֶׁל בִּנְיָין רַב קוֹמוֹת, שָׁם הֵם פָּגְשׁוּ אֶת שׁוּלִי הַשְּׁכֵנָה שֶׁזָּכְתָה לְהִתְיַידֵּד אִיתָּם וְיַחַד הֵם מִתְגַּבְּרִים עַל הַמְּזִימּוֹת שֶׁל נַפְתָּלִי הַדַּיָּיר הַמְּרוּשָּׁע מֵהַמַּרְתֵּף.
דְּמוּיוֹת
דְּמוּיוֹת רָאשִׁיּוֹת
אִיץ (רוּת קַפְּלָן רוֹט) – חַיְיזָרִית וְרוּדָּה, הִגִּיעָה יַחַד עִם אִיץ מִכּוֹכַב "הָפוּכוּלוּס" לְכַדּוּר הָאָרֶץ.
מִיץ (אִילָן סָבִיר) – חַיְיזָר כָּתוֹם, הִגִּיעַ יַחַד עִם אִיץ מִכּוֹכַב "הָפוּכוּלוּס" לְכַדּוּר הָאָרֶץ.
שׁוּלִי (חַגִּית שֶׁמְלִי) – שְׁכֵנָה אֲשֶׁר עָלְתָה לַעֲלִיַּית הַגַּג וְהִתְיַידְּדָה עַם אִיץ וּמִיץ.
כַּרְמֶלָה (חַגִּית דַּסְבֵּרְג) - שְׁכֵנָה אֲשֶׁר לֹא הֶאֱמִינָה שֶׁיֵּשׁ חַיְיזָרִים בָּעוֹלָם.
נַפְתָּלִי (אִיצ'וֹ אֲבִיטָל) - דַּיָּיר מֵהַמַּרְתֵּף שֶׁזּוֹמֵם מְזִימּוֹת מְרוּשָּׁעוֹת.
דְּמוּיוֹת אוֹרְחוֹת:
בְּאֵרִי (אִילָן סָבִיר) - חַיְיזָר בַּעַל אַף אָרוֹךְ, הִגִּיעַ לְהִתְאָרֵחַ מִכּוֹכַב "אַפַּאכְרוֹס".
 אִרְגּוּנֵי חֶסֶד בְּיִשְׂרָאֵל
אִרְגּוּנֵי חֶסֶד הוּא שֵׁם כּוֹלֵל לְאִרְגּוּנִים שֶׁפּוֹעֲלִים לְמַעַן מַתַּן עֶזְרָה בְּאוֹפֶן ווֹלוּנְטָרִי, מִמְּנִיעִים אַ-פּוֹלִיטִיִּים, סוֹצְיוֹלוֹגִיִּים אוֹ מִגְדָּרִיִּים, לָרוֹב מִתּוֹךְ אִידֵאוֹלוֹגְיָה שֶׁל צְדָקָה וּגְמִילוּת חֲסָדִים. בְּיִשְׂרָאֵל פּוֹעֲלִים אִרְגּוּנִים הִתְנַדְּבוּתִיִּים רַבִּים הַפּוֹעֲלִים לְמַעַן מַטְּרוֹת צְדָקָה מְגֻוָּונוֹת: מַתַּן אוֹכֶל לִרְעֵבִים, אִיסּוּף צַעֲצוּעִים לִילָדִים עֲנִיִּים, חֲלוּקָּה שֶׁל בְּגָדִים לְנִזְקָקִים וְעוֹד. חֵלֶק נִיכָּר מֵהָאִרְגּוּנִים אֵינוֹ נִתְמָךְ עַל יְדֵי הַמֶּמְשָׁלָה.
בְּיִשְׂרָאֵל יֶשְׁנָן כַּ-15 אֶלֶף עֲמוּתּוֹת פְּעִילוֹת[דָּרוּשׁ מָקוֹר], כְּ-16 אָחוּז מֵהֶן עוֹסְקוֹת בִּרְוָוחָה וּבְרִיאוּת.
בְּאוֹפֶן מָסוֹרְתִּי, בִּתְחוּם הָרְוָוחָה פּוֹעֲלִים אִרְגּוּנִים דָּתִיִּים רַבִּים וְאֵלֶּה מִתְרַכְּזִים בְּסִיּוּעַ כַּסְפִּי וְחוֹמְרִי, הֶמְשֵׁךְ לְתַרְבּוּת הַצְּדָקָה שֶׁאִפְיְינָה אֶת הַחֶבְרָה הַיְּהוּדִית גַּם בַּגּוֹלָה. בְּדֶרֶךְ כְּלָל תּוֹרְמִים הָאִרְגּוּנִים הַדָּתִיִּים כְּסָפִים אוֹ מוּצָרִים לִבְנֵי אָדָם, בְּעוֹד שֶׁאִרְגּוּנִים לֹא דָּתִיִּים בִּתְחוּם הָרְוָוחָה נָטוּ לְהָקִים אִרְגּוּנֵי פִילַנְתְּרוֹפְּיָה, שֶׁנָּתְנוּ כְּסָפִים לְקִידּוּם מַטָּרוֹת כְּמוֹ חִינּוּךְ וּבְרִיאוּת.
בַּמִּגְזָר הָעַרְבִי בּוֹלֵט חֶסְרוֹנָן שֶׁל עֲמוּתּוֹת הַמְּטַפְּלוֹת בִּמְצוּקָה חוֹמְרִית בַּשָּׁנִים הָאַחֲרוֹנוֹת, עַל אַף הַמְּצוּקָה הַכַּלְכָּלִית הַקָּשָׁה בְּמִגְזָר זֶה. פְּרוֹפ' בִּנְיָמִין גִּדְרוֹן, מְנַהֵל הַמֶּרְכָּז הַיִּשְׂרְאֵלִי לְחֵקֶר הַמִּגְזָר הַשְּׁלִישִׁי, מְשַׁעֵר שֶׁיִּיתָּכֵן שֶׁאִרְגּוּנִים אִיסְלָאמִיִּים שֶׁאֵינָם רְשׁוּמִים כַּעֲמוּתּוֹת מְמַלְּאִים חֵלֶק מֵהַחֶסֶר[1].
אִרְגּוּנֵי חֶסֶד בּוֹלְטִים
לְהַלָּן רְשִׁימַת אִרְגּוּנֵי חֶסֶד שֶׁמַּחֲזוֹר הַפְּעִילוּת הַשְּׁנָתִית שֶׁלָּהֶם בִּשְׁנַת 2010 (דָּרוּשׁ רִיעֲנוּן) עָלָה עַל 10 מִילְיוֹן ש"ח, לְפִי הַדִּיוּוּחַ בַּאֲתַר גַּיְידְסְטָאר.
 "דֶּלֶק" חֶבְרַת הַדֶּלֶק הַיִּשְׂרְאֵלִית
"דֶּלֶק" חֶבְרַת הַדֶּלֶק הַיִּשְׂרְאֵלִית בע"מ, הַיְּדוּעָה יוֹתֵר בַּשֵּׁם דֶּלֶק, הִיא אַחַת מֵאַרְבַּע חֶבְרוֹת הַדֶּלֶק הַגְּדוֹלוֹת בְּיִשְׂרָאֵל. בְּבַעֲלוּתָהּ מְאוֹת תַּחֲנוֹת דֶּלֶק בְּרַחֲבֵי הָאָרֶץ, בְּחֶלְקָן גַּם חֲנוּיוֹת נוֹחוּת תַּחַת הַמּוּתָג "מֶנְטָה" וּמִתְחָמִים קִמְעוֹנָאִיִּים. בְּבַעֲלוּת הַחֶבְרָה מִסְפַּר חֲבָרוֹת-בָּנוֹת בִּתְחוּמֵי הָאֵנֶרְגִּיָּה וְהַלּוֹגִיסְטִיקָה, הַמְּסַפְּקוֹת שֵׁירוּתֵי שִׁינּוּעַ וְהוֹבָלָה שֶׁל דֶּלֶק, מַעֲרָכוֹת לוֹגִיסְטִיּוֹת, יִיצּוּר שְׁמָנִים וְחוֹמְרֵי אִיטּוּם, וְשֵׁירוּתֵי נִיפּוּק לָאוֹנִיּוֹת בִּנְמֵלִים בָּאָרֶץ וּבָעוֹלָם.
הִיסְטוֹרְיָה
הַחֶבְרָה הוּקְמָה עַל יְדֵי מֶמְשֶׁלֶת יִשְׂרָאֵל בְּדֵצֶמְבֶּר 1951, וּבְדוֹמֶה לְחֶבְרוֹת הָאֵנֶרְגִּיָּה הָאֲחֵרוֹת (פָּז וְסוֹנוֹל) הָיְיתָה נְתוּנָה בָּעֲשׂוֹרִים הָרִאשׁוֹנִים לְפִיקּוּחַ מֶמְשַׁלְתִּי. עַד 1960 הִתְרַחֲבָה הַחֶבְרָה וּפָרְשָׂה 44 תַּחֲנוֹת דֶּלֶק בְּרַחֲבֵי הָאָרֶץ; בְּמַקְבִּיל הִצְטַמְצֵם חֶלְקָן בַּשּׁוּק שֶׁל חֶבְרוֹת הַדֶּלֶק הַזָּרוֹת, שֶׁפָּעֲלוּ בָּאָרֶץ מֵאָז תְּקוּפַת הַמַּנְדָּט הַבְּרִיטִי. בִּתְקוּפָה זוֹ נוֹסְדָה גַּם חֶבְרַת הַשְּׁמָנִים "דֵּלֵקוֹל". בְּ-1988 בּוּצְּעָה רֵפוֹרְמָה בְּשׁוּק הַדֶּלֶק בָּאָרֶץ, וְהַמֶּמְשָׁלָה הִפְחִיתָה אֶת הַפִּיקּוּחַ עַל הַשּׁוּק בְּאוֹפֶן מַשְׁמָעוּתִי; בְּעִקְבוֹת זֹאת גָּבְרָה הַתַּחֲרוּת בֵּין שְׁלוֹשׁ חֶבְרוֹת הַדֶּלֶק הַגְּדוֹלוֹת.
בְּ-1998 רָכַשׁ אֶת הַשְּׁלִיטָה בַּחֶבְרָה אִישׁ הָעֲסָקִים יִצְחָק תְּשׁוּבָה, וְהַחֶבְרָה הָפְכָה לְחֵלֶק מִקְּבוּצַת דֶּלֶק שֶׁבִּשְׁלִיטָתוֹ, חֶבְרַת אַחְזָקוֹת מֵהַגְּדוֹלוֹת בְּיִשְׂרָאֵל. בִּתְחִילַּת שְׁנוֹת הָ-2000 יָצְאָה הַחֶבְרָה עִם מִיתּוּג חָדָשׁ וְהֵחֵלָּה לִפְרוֹשׂ אֶת רֶשֶׁת חֲנוּיוֹת הַנּוֹחוּת "מֶנְטָה" בְּחֵלֶק מִתַּחֲנוֹת הַדֶּלֶק שֶׁלָּהּ. בְּיוּנִי 2007 זָכְתָה הַחֶבְרָה בְּמִכְרָז לְהַפְרָטַת מְסוֹפֵי הַנִּיפּוּק וְהָאִחְסוּן שֶׁל פִּי גְּלִילוֹת בְּאַשְׁדּוֹד, בְּאֵר-שֶׁבַע וִירוּשָׁלַיִם; הָעִסְקָה, בְּשֹׁוִוי 800 מִילְיוֹן ש"ח, אִפְשְׁרָה לַחֶבְרָה כּוֹשֶׁר אִחְסוּן שֶׁל 600 מִילְיוֹן לִיטֶר (40% מִכּוֹשֶׁר הָאִחְסוּן שֶׁל מֶשֶׁק הַדֶּלֶק בָּאָרֶץ) וְכוֹשֶׁר נִיפּוּק הַמַּסְפִּיק לִצְרִיכַת הַדֶּלֶק הַכּוֹלֶלֶת שֶׁל הַמְּדִינָה, דָּבָר שֶׁהֶעֱנִיק לַחֶבְרָה יִתְרוֹן מַשְׁמָעוּתִי בִּתְחוּם יִיבּוּא הַדְּלָקִים בָּאָרֶץ. בְּמַקְבִּיל רָכְשָׁה חֶבְרַת "דֶּלֶק אֵירוֹפָּה", שֶׁבָּהּ שׁוּתָּפָה חֶבְרַת דֶּלֶק, 869 תַּחֲנוֹת דֶּלֶק (כּוֹלֵל 400 חֲנוּיוֹת נוֹחוּת), בְּהוֹלַנְד, לוּקְסֶמְבּוּרְג וּבֶלְגִּיָּה, מִיְּדֵי חֶבְרַת הָאֵנֶרְגִּיָּה "שֶׁבְרוֹן טֶקְסָקוֹ".
בְּאוֹגוּסְט 2007 הוּנְפְּקוּ לָרִאשׁוֹנָה מְנָיוֹת הַחֶבְרָה בַּבּוּרְסָה לִנְיָירוֹת עֵרֶךְ בְּתֵל אָבִיב, וְדֶלֶק הָפְכָה לְחֶבְרָה צִיבּוּרִית, וּבַהֶמְשֵׁךְ נִכְלְלָה בְּמַדַּד תֵּל אָבִיב 100.
בְּאַפְּרִיל 2013 רָכְשָׁה "דֶּלֶק פֶּטְרוֹלְיוּם", חֶבְרַת הַבַּת שֶׁל קְבוּצַת דֶּלֶק שֶׁדַּרְכָּהּ הֶחְזִיקָה זוֹ בְּחֶבְרַת "דֶּלֶק", אֶת יִתְרַת 12.1% הַמְּנָיוֹת שֶׁהָיוּ בִּידֵי הַצִּיבּוּר, וְדֶלֶק הָפְכָה לְחֶבְרָה פְּרָטִית בְּבַעֲלוּת מְלֵאָה שֶׁל דֶּלֶק פֶּטְרוֹלְיוּם, וּמְנָיוֹת הַחֶבְרָה נִמְחֲקוּ מֵהַבּוּרְסָה בת"א[1].
כְּשָׁנָה לְאַחַר מִכֵּן מָחֲקָה קְבוּצַת דֶּלֶק אֶת חֶבְרַת דֶּלֶק פֶּטְרוֹלְיוּם כְּחֵלֶק מִתַּהֲלִיךְ "שִׁיטּוּחַ פִּירָמִידָה", כָּךְ שֶׁלְּאַחַר תַּהֲלִיךְ זֶה הָפְכָה דֶּלֶק לְחֶבְרָה בְּבַעֲלוּת מְלֵאָה וִישִׁירָה שֶׁל קְבוּצַת דֶּלֶק[2].
דֶּגֶל הַחֶבְרָה
בַּכְּנִיסָה לְכׇל תַּחֲנוֹת הַדֶּלֶק שֶׁל הַחֶבְרָה מוּנָף דֶּגֶל הַחֶבְרָה. הַדֶּגֶל, הַקּוֹדֵם וְהַנּוֹכְחִי כְּאֶחָד, הוּא מְקוֹרִי בְּעִיצּוּבוֹ וּבְכָךְ בּוֹלֵט בְּאוֹפֶן חָרִיג מִשְּׁאַר הַדְּגָלִים הַמִּסְחָרִיִּים בָּאָרֶץ
[3].
אֶפְרַיִם אׇשְׁרִי
הָרַב אֶפְרַיִם אׇשְׁרִי (מְבוּטָּא, וּלְעִיתִּים אַף נִכְתָּב כְּאוֹשְׁרִי; ה'תרע"ד, 1914 - 28 בְּסֶפְּטֶמְבֶּר 2003) הָיָה תַּלְמִיד חָכָם, פּוֹסֵק וְרֹאשׁ יְשִׁיבָה. עָסַק רַבּוֹת בַּשּׁוֹאָה, וְחִיבֵּר סֵפֶר שׁוּ"ת שֶׁעָסַק בִּשְׁאֵלוֹת הִלְכָתִיּוֹת שֶׁנָּגְעוּ לַשּׁוֹאָה, בַּשֵּׁם מִמַּעֲמַקִּים.
בִּיּוֹגְרַפְיָה
אׇשְׁרִי נוֹלַד בְּקוּפִּישׁוֹק שֶׁבְּלִיטָא לָרַב דּוֹב אׇשְׁרִי. הִתְחַנֵּךְ בִּישִׁיבַת סְלוֹבּוֹדְקָה, בִּישִׁיבַת פּוֹנִיבֶז' וּבְתַלְמוּד תּוֹרָה בְּקֵלְם. בִּשְׁנַת ה'תרצ"ד (1934) נִסְמַךְ לְהוֹרָאָה בִּידֵי רַבָּהּ שֶׁל קוֹבְנָה, הָרַב אַבְרָהָם דּוֹב כָּהֲנָא שַׁפִּירָא וּבִידֵי רַבִּי לֵייבּ רוּבִּין מִוִּוילְקוֹמִיר, רַבִּי עֶזְרָא אַלְטְשְׁוּלֶר, ר' אַבְרָהָם זְאֵב הֵלֶר מִמָּרִיאַמְפּוֹלֶה וְר' בָּרוּךְ הוֹרוֹבִיץ מֵאֵלֶקְסוֹט. לְאַחַר מִכֵּן שִׁימֵּשׁ כְּרַב בִּסְלוֹבּוֹדְקָה וְלִימֵּד בְּבֵית מִדְרָשׁ מְקוֹמִי.[1]
בְּמֶשֶׁךְ מִלְחֶמֶת הָעוֹלָם הַשְּׁנִיָּיה שָׁהָה הָרַב אׇשְׁרִי בְּגֵטוֹ קוֹבְנָה, שָׁם שִׁימֵּשׁ יַד יְמִינוֹ שֶׁל רַב הָעִיר - הָרַב אַבְרָהָם דּוֹב כָּהֲנָא שַׁפִּירָא - וְעָמַד בְּרֹאשׁ הָאִרְגּוּן "תִּפְאֶרֶת בַּחוּרִים" בְּגֵטוֹ קוֹבְנָה. בִּתְקוּפָה זוֹ רוֹב רַבָּנֵי הַגֵּטוֹ נִרְצְחוּ בִּידֵי הַנָּאצִים וְהַלִּיטָאִים, וְהָרַב שַׁפִּירָא עַצְמוֹ כְּבָר הָיָה חוֹלֶה וְחַלָּשׁ וְנִבְצַר מִמֶּנּוּ לְתַפְקֵד כְּפוֹסֵק בְּאוֹפֶן מָלֵא, וְעַל כֵּן הָרַב שַׁפִּירָא בִּיקֵּשׁ מֵהָרַב אׇשְׁרִי שֶׁיַּעֲנֶה עַל שְׁאֵלוֹת הִלְכָתִיּוֹת בִּמְקוֹמוֹ. אֶת הַדַּפִּים בָּהֶם נִכְתְּבוּ הַתְּשׁוּבוֹת שָׁמַר הָרַב אׇשְׁרִי טְמוּנִים בָּאֲדָמָה, וְרַק לְאַחַר הַמִּלְחָמָה קִיבֵּץ אוֹתָם לְסֵפֶר שְׁאֵלוֹת וּתְשׁוּבוֹת "מִמַּעֲמַקִּים".
לְאַחַר הַמִּלְחָמָה הָרַב אׇשְׁרִי נוֹתַר הָרַב הַיָּחִיד בְּקוֹבְנָה, וְהִמְשִׁיךְ שָׁם בִּפְעִילוּת צִיבּוּרִית לְמַעַן הַפְּלִיטִים. מְאוּחָר יוֹתֵר הוּא הִיגֵּר לְאוֹסְטְרִיָּה וְהֵקִים יְשִׁיבָה לְיַד זַלְצְבּוּרְג, וּמִשָּׁם לְרוֹמָא, שָׁם שִׁימֵּשׁ כְּרֹאשׁ יְשִׁיבַת "מְאוֹר הַגּוֹלָה". לְאַחַר תְּקוּפָה זוֹ הִיגֵּר לְאַרְצוֹת הַבְּרִית, שָׁם גַּם פִּרְסֵם אֶת סְפָרָיו. בְּמֶשֶׁךְ עֶשְׂרוֹת שָׁנִים הִנְהִיג אֶת קְהִילַּת בֵּית הַמִּדְרָשׁ הַגָּדוֹל בְּלוֹאֵר אִיסְט סַיְיד שֶׁל מַנְהֵטְן בִּנְיוּ יוֹרְק.
הָרַב אׇשְׁרִי נִפְטַר בְּב' תִּשְׁרֵי שְׁנַת ה'תשס"ד, 2003, בִּנְיוּ יוֹרְק. לְאַחַר הַלְוָויָה מְכוּבֶּדֶת בְּחוּ"ל, הוּא הוֹעֳלָה לִקְבוּרָה בִּירוּשָׁלַיִם.
פְּסִיקוֹתָיו
הַגֵּטוֹ זִימֵּן לָרַב אׇשְׁרִי שְׁאֵלוֹת לֹא שִׁגְרָתִיּוֹת, וְכֵן הֵבִיא נוֹשְׂאִים תֵּאוֹרֵטִיִּים שׁוֹנִים לִידֵי מַשְׁמָעוּת מַעֲשִׂית. כָּךְ לְדוּגְמָה הוּא נִשְׁאַל הַאִם נִיתָּן לְבָרֵךְ בְּבִרְכוֹת הַשַּׁחַר בִּרְכַּת "שֶׁלֹּא עָשַׂנִי עָבֶד", מִכֵּיוָון שֶׁהַצְהָרָה זוֹ עָמְדָה בִּסְתִירָה לְמַצַּב הַיְּהוּדִים בַּגֵּטוֹ. בִּתְשׁוּבָתוֹ כּוֹתֵב הָרַב אׇשְׁרִי: "עִיקַּר הַבְּרָכָה הַזֹּאת נִתְקְנָה לֹא עַל עַבְדוּת הַגּוּף, אֶלָּא עַל עַבְדוּת הַנֶּפֶשׁ... וְאַדְּרַבָּה, דַּוְוקָא כָּעֵת חוֹבָה עָלֵינוּ לְבָרֵךְ בְּרָכָה זֹאת לְמַעַן יַכִּירוּ אוֹיְבֵינוּ וּמְנַדֵּינוּ, שֶׁלַּמְרוֹת שֶׁאָנוּ נְתוּנִים בְּיָדָם לַעֲשׂוֹת בָּנוּ כִּרְצוֹנָם הָרַע, בְּכׇל זֹאת רוֹאִים אָנוּ אֶת עַצְמֵנוּ כִּבְנֵי חוֹרִין הַנְּתוּנִים בְּמָצוֹר וּבְשִׁבְיָה, וּקְרוֹבָה יְשׁוּעָתֵנוּ לָבוֹא."[2]
עִנְיָין הִלְכָתִי נוֹסָף שֶׁנּוֹבֵעַ מֵהַשּׁוֹאָה, הוּא הַאִם יֵשׁ לוֹמַר קַדִּישׁ עַל גּוֹי שֶׁנֶּחְשַׁב לַחֲסִיד אוּמּוֹת הָעוֹלָם. הָרַב אׇשְׁרִי כּוֹתֵב כִּי הַדָּבָר אֵינוֹ רַק מוּתָּר, אֶלָּא גַּם בַּעַל עֵרֶךְ סְגוּלִּי - מִצְוָוה.[3]
 אַמְבָּה אָלָגִי
אַמְבָּה אָלָגִי הוּא אַמְבּ (מִבְצָר הֲרָרִי הַמִּסְתַּמֵּךְ עַל תְּצוּרָה גֵּאוֹלוֹגִית שְׁכִיחָה בְּאֶתְיוֹפְּיָה, בְּצוּרַת הַר שׁוּלְחָן הַמּוּקָּף בִּמְצוּקִים תְּלוּלִים), בִּצְפוֹן אֶתְיוֹפְּיָה, בִּמְחוֹז תִּיגְרַאי. הָאַמְבּ שׁוֹלֵט עַל הַדֶּרֶךְ בֵּין הָעִיר מֵקֵלֶה דָּרוֹמָה לָעִיר מָאִיצ'וֹ. בְּשֶׁל מִיקּוּמוֹ הָאַסְטְרָטֵגִי הָיָה הָאַמְבּ זִירָה לְמִסְפַּר קְרָבוֹת חֲשׁוּבִים בַּהִיסְטוֹרְיָה הָאֶתְיוֹפִּית.
הִיסְטוֹרְיָה
אִזְכּוּרִים רִאשׁוֹנִים
הַפַּעַם הָרִאשׁוֹנָה בָּהּ מוֹפִיעַ שְׁמוֹ שֶׁל הָהָר בַּכְּתוּבִים הָיְיתָה בְּמַהֲלַךְ הַמֵּאָה הַ-15 כַּאֲשֶׁר קֵיסַר אֶתְיוֹפְּיָה בָּאֵדָה מַרְיָאם הָרִאשׁוֹן הוֹבִיל מַסָּע צְבָאִי כְּנֶגֶד שֵׁבֶט בְּשֵׁם דּוֹבֵּאָה שֶׁשָּׁלַט בָּאֵזוֹר וְהִפְרִיעַ לִנְתִיבֵי הַתַּחְבּוּרָה. בִּתְחִילַּת הַמֵּאָה הַ-19 הִשְׁתַּמֵּשׁ הָרָאס ווֹלְדֶה סֵלָאסִי בַּמְּצוּדָה שֶׁעַל הָהָר עַל מְנַת לִכְלוֹא אֶת גֶּבְּרֶה גּוּרוֹ, אָחִיו שֶׁל סַבָּגָדִיס שֶׁמָּרַד כְּנֶגְדּוֹ. הַדֶּרֶךְ שֶׁסָּלְלוּ הַבְּרִיטִים בְּמַהֲלַךְ הַמִּשְׁלַחַת הַצְּבָאִית לְאֶתְיוֹפְּיָה (1868), מֵהַחוֹף אֶל אַמְבָּה מַרְיָאם, עָבְרָה לְיַד אַמְבָּה אָלָגִי. בְּפֶבְּרוּאָר 1889 לָחַם רָאס אָלוּלָה אֶנְגְדָה בְּרָאס סֵיוּם גֶּבְּרֶה, וּבִקְרָב זֶה נִפְצַע אֶנְגְדָה.
קְרַב אַמְבָּה אָלָגִי (1895)
עֵרֶךְ מוּרְחָב – קְרַב אַמְבָּה אָלָגִי (1895)
הַקְּרָב הִתְרַחֵשׁ בְּמַהֲלַךְ הַמִּלְחָמָה הָאִיטַלְקִית-אֶתְיוֹפִּית הָרִאשׁוֹנָה בְּדֵצֶמְבֶּר 1895. בְּ-7 בְּדֵצֶמְבֶּר הֵבִיסוּ הָאֶתְיוֹפִּים כּוֹחַ אִיטַלְקִי שֶׁל 2,350 חַיָּילִים שֶׁהוּבַל עַל יְדֵי פִּיֶּיטְרוֹ טוֹסֵלִי, וּבִ-8 בְּדֵצֶמְבֶּר, גֵּירֵשׁ הַקֵּיסָר הָאֶתְיוֹפִּי מֵנֵלִיק הַשֵּׁנִי אֶת כּוֹחוֹתָיו שֶׁל הַגֵּנֵרָל הָאִיטַלְקִי אוֹרֶסְטֶה בָּרָטִיֵּירִי. לְאַחַר הַתְּבוּסָה נָסוֹגוּ הָאִיטַלְקִים מֵאַמְבָּה אָלָגִי וּמִגּוֹנְדָר כּוּלָּהּ, לְמִבְצַר מֵקֵלֶה שֶׁעָלָיו צָר מֵנֵלִיק בְּמֶשֶׁךְ חֲמִישָּׁה עָשָׂר יָמִים.
קְרַב אַמְבָּה אָלָגִי (1941)
הַקְּרָב הַשֵּׁנִי שֶׁל אַמְבָּה אָלָגִי נֶעֱרַךְ בְּמִסְגֶּרֶת הַמַּעֲרָכָה בְּמִזְרַח אַפְרִיקָה בְּמִלְחֶמֶת הָעוֹלָם הַשְּׁנִיָּיה. מִתְּחִילַּת שְׁנַת 1941 פָּתַח הַצָּבָא הַבְּרִיטִי בְּאֶמְצָעוּת יְחִידוֹת מִצְּבָא הוֹדּוּ הַבְּרִיטִית וְעִמָּם כּוֹחוֹת מִצְּבָא דְּרוֹם אַפְרִיקָה, צְבָא צָרְפַת הַחוֹפְשִׁית וִיחִידוֹת נוֹסָפוֹת שֶׁל לוֹחֲמֵי גֵּרִילָה אֶתְיוֹפִּים הַמְּכוּנִּים כּוֹחַ גִּדְעוֹן בְּמַעֲרָכָה כְּנֶגֶד הָאִיטַלְקִים שֶׁהֶחְזִיקוּ בְּאֶתְיוֹפְּיָה, סוֹמַלְיָה וְאָרִיתְרֵיאָה. בִּתְנוּעַת מֶלְקָחַיִים אִיגְּפוּ כּוֹחוֹתָיו שֶׁל הַגֵּנֵרָל וִילְיָאם פְּלָאט אֶת הָאִיטַלְקִים מִצָּפוֹן, וְכוֹחוֹתָיו הַדְּרוֹם אַפְרִיקָאִים שֶׁל הַגֵּנֵרָל אָלָן קַנִינְגְהָם אֶת הָאִיטַלְקִים מִדָּרוֹם. הָאִיטַלְקִים נָסוֹגוּ וְהִתְבַּצְּרוּ בְּמִבְצַר אַמְבָּה אָלָגִי, שָׁם חָצְבוּ מְצוּדָה בְּמִנְהָרוֹת בָּאֶבֶן הַטִּבְעִית. תַּחַת פִּיקּוּדוֹ שֶׁל הַמִּשְׁנֶה לַמֶּלֶךְ הָאִיטַלְקִי, הַנָּסִיךְ אָמֵדֵאוֹ הַדּוּכָּס הַשְּׁלִישִׁי מֵאָאוֹסְטָה נִיהֲלוּ הָאִיטַלְקִים בַּמָּקוֹם קְרַב מִגְנָנָה. בִּתְחִילַּת מַאי הֵחֵלּוּ הַבְּרִיטִים לִתְקוֹף מִצָּפוֹן, וְהִקִּיפוּ אֶת הָהָר מִצִּידּוֹ הַמִּזְרָחִי וְהַמַּעֲרָבִי. בְּ-14 בְּמַאי הִצְטָרְפוּ אֲלֵיהֶם הַכּוֹחוֹת הַדְּרוֹם אַפְרִיקָאִים מִדָּרוֹם, וְהָהָר הוּקַּף מִכׇּל עֲבָרָיו. הַבְּרִיטִים תִּכְנְנוּ לִתְקוֹף אֶת הַמְּצוּדָה בְּאוֹפֶן חֲזִיתִי, אַךְ פָּגָז שֶׁפָּגַע פְּגִיעָה מְמוּזֶּלֶת בְּמִצְבּוֹר דֶּלֶק, הֵבִיא לְזִיהוּם מֵי הַשְּׁתִיָּיה בְּדֶלֶק, וְלִכְנִיעַת מְגִנֵּי הָהָר בַּ-15 בְּמַאי.
קְרַב אַמְבָּה אָלָגִי הַשְּׁלִישִׁי
בְּ-1943 הִתְקַיֵּים לְיַד אַמְבָּה אָלָגִי הַקְּרָב הָאַחֲרוֹן בְּ"מֶרֶד ווֹיָאנֶה" שֶׁכֻּוַּון כְּנֶגֶד שִׁלְטוֹנוֹ שֶׁל הַיְילֶה סֵלָאסִי. הַמּוֹרְדִים כָּבְשׁוּ אֶת מֵקֵלֶה וְהִמְשִׁיכוּ דָּרוֹמָה אֶל הָהָר, בּוֹ נִתְקְלוּ בַּעֲשֶׂרֶת אֲלָפִים חַיָּילִים שֶׁל מֶמְשֶׁלֶת אֶתְיוֹפְּיָה. בֵּין 18 בְּסֶפְּטֶמְבֶּר לְ-6 בְּאוֹקְטוֹבֶּר 1943 נִיסּוּ הַמּוֹרְדִים, בְּהַנְהָגַת בְּלָאטָה הַיְילֶה מַרְיָאם רֵדָה לִכְבּוֹשׁ אֶת עֶמְדוֹת חַיָּילֵי הַמֶּמְשָׁלָה בָּהָר, אַךְ לֹא עָמְדוּ בִּפְנֵי הַפְגָּזוֹת הָאַרְטִילֶרְיָה שֶׁל הַצָּבָא הַבְּרִיטִי שֶׁסִּייֵּעַ לְסֵלָאסִי, וּבְסוֹפוֹ שֶׁל דָּבָר הִתְפַּזְּרוּ.
 אַ גֵעשֵׁעפְט
אַ גֵעשֵׁעפְט (בְּעִבְרִית: הָעִסְקָה) הוּא שְׁמוֹ שֶׁל סֶרֶט פְּעוּלָּה בְּיִידִישׁ מִשְּׁנַת 2005, שֶׁהוּפַק, צוּלַּם, וּבוּיַּם עַל יְדֵי יְהוּדִים חֲרֵדִים בְּמוֹנְסִי, נְיוּ יוֹרְק. זֶהוּ הַסֶּרֶט הָרִאשׁוֹן בְּיִידִישׁ שֶׁיּוֹצֵא מִקֶּרֶב חוּגִים חֲרֵדִיִּים, וְאַחַד הַסְּרָטִים הָרִאשׁוֹנִים שֶׁהוּפְקוּ בְּיִידִישׁ מֵאָז שְׁנוֹת הַ-40.
הַסֶּרֶט צוּלַּם בְּאַרְצוֹת הַבְּרִית בְּרָמַת הֲפָקָה גְּבוֹהָה יַחֲסִית, וְזָכָה שָׁם לִתְהוּדָה. כְּמוֹ כׇּל סֶרֶט חֲרֵדִי טִיפּוּסִי, צֶוֶות הַשַּׂחְקָנִים בַּסֶּרֶט מוּרְכָּב מִגְּבָרִים בִּלְבַד.
"אַ גֵעשֵׁעפְט" הוּא סֶרֶט שׁוֹד, הַמְּגוֹלֵל אֶת סִיפּוּרוֹ שֶׁל פֶּרֶץ, סוֹחֵר שֶׁחַיָּיו מִשְׂתָּרְעִים עַל קַו הַתֶּפֶר בֵּין הָרֶקַע הַחֲרֵדִי שֶׁלּוֹ לְבֵין חַיֵּי עוֹלָם תַּחְתּוֹן. פָּרֶץ מוּכָן לְהִשְׁתַּמֵּשׁ בְּאַלִּימוּת כְּדֵי לְנַהֵל אֶת עֲסָקָיו הַמְּפוּקְפָּקִים, וְהַסֶּרֶט עוֹקֵב אַחַר הַקַּרְיֵירָה הַפְּלִילִית שֶׁלּוֹ. בַּמַּאי הַסֶּרֶט, חֲסִיד וִיזְ'נִיץ, צִייֵּן שֶׁהוּא הוּשְׁפַּע מִכַּמָּה סִרְטֵי מֵיינְסְטְרִים אָמֵרִיקָאִיִּים, כּוֹלֵל "לֵב אַמִּיץ" שֶׁל הַבַּמַּאי מֵל גִיבְּסוֹן. עוֹרֵךְ הַסֶּרֶט, רוֹלַנְד מִילְמָן, עָרַךְ מִסְפַּר סִרְטֵי יִידִישׁ (סֶרֶט כּוֹשֶׁר "שְׁוַויְיץ" וְסֶרֶט בִּישּׁוּל "נוֹ שְׁמַאלְץ"), וּלְפִיו הַמִּשְׂחָק בַּסֶּרֶט "לֹא נִפְלָא, אֲבָל הַצִּילּוּם לֹא רַע".
סְבִיב הַשִּׁיוּוּק הַוִּירָאלִי וְהָאִינְטֶרְנֵטִי שֶׁל הַסֶּרֶט, שֶׁהוּפְנָה בְּעִיקָּר לְקָהָל חֲרֵדִי, הִתְעוֹרֵר פּוּלְמוּס כַּאֲשֶׁר רַבָּנִים אָסְרוּ לִצְפּוֹת בּוֹ בְּשֶׁל חֲשָׁשָׁם מֵחֲדִירָה שֶׁל הַשְׁפָּעוֹת הַקּוֹלְנוֹעַ וְהַתֵּיאַטְרוֹן לַצִּיבּוּר הַחֲרֵדִי. הַסּוֹצְיוֹלוֹג חַיִּים וַקְסְמָן מֵאוּנִיבֶרְסִיטַת רַאטְגֵרְס שִׁיעֵר כִּי הָעוּבְדָּה שֶׁהַסֶּרֶט נַעֲשָׂה בְּיִידִישׁ הוֹסִיפָה לַחֲשָׁשׁוֹת, מִשּׁוּם שַׁ"אֲנָשִׁים עֲלוּלִים לַחֲשׁוֹב אִם זֶה בְּיִידִישׁ, זֶה עוֹשֶׂה אֶת זֶה כָּשֵׁר". עַל-פִּי וַקְסְמָן, הַהוֹפָעָה שֶׁל סֶרֶט בְּיִידִישׁ הָיָה חֵלֶק מִשִּׁינּוּי מַשְׁמָעוּתִי בָּעוֹלָם הַחֲרֵדִי אֲשֶׁר הֵחֵל לְאַמֵּץ מִגְוָון שֶׁל מוּצְרֵי תַּרְבּוּת, כּוֹלֵל מוּזִיקָה, סִיפּוֹרֶת, וַאֲפִילּוּ תֵּיאַטְרוֹן, שֶׁהַדּוֹרוֹת הַקּוֹדְמִים לָהֶם נִמְנְעוּ מֵהֶם.[1]
הַסֶּרֶט מָשַׁךְ גַּם תְּשׂוּמַת לֵב בְּחוּגִים יְהוּדִיִּים חִילּוֹנִיִּים בַּעֲלֵי עִנְיָין בְּיִידִישׁ, מִשּׁוּם שֶׁמְּדוּבָּר בְּמִסְמָךְ בְּיִידִישׁ בֶּן זְמַנֵּנוּ.[2] הוּא הוּקְרַן בֵּין הַיֶּתֶר בְּ"אַשְׁכְּנַז פֶסְטִיבָל" בְּטוֹרוֹנְטוֹ וּבְפֶסְטִיבַל הַיִּידִישׁ בְּאוּנִיבֶרְסִיטַת אָרִיזוֹנָה.[3] בִּשְׁנַת 2009 יָצָא הַסֶּרֶט בְּיִשְׂרָאֵל תַּחַת הַשֵּׁם "הָעִסְקָה", וְצוֹרַף לוֹ דִּיבּוּב בְּעִבְרִית.
 אוֹפֶן חִיתּוּךְ
אוֹפֶן חִיתּוּךְ הוּא מִידַּת הַחֲסִימָה שֶׁל מַעֲבַר הָאֲוִויר בִּבְסִיס הַחִיתּוּךְ שֶׁהוּא הַמָּקוֹם בּוֹ מִסְפַּר אֵיבְרֵי הֲגִיָּיה סוֹתְמִים בְּאוֹפֶן מָלֵא אוֹ חֶלְקִי אֶת חֲלַל הַפֶּה אוֹ הַגָּרוֹן וְגוֹרְמִים לַהֲפָקַת הָעִיצּוּרִים כְּתוֹצָאָה מִתְּזוּזָתָם שֶׁל אֵיבְרֵי הֲגִיָּיה מְסוּיָּמִים כְּלַפֵּי אֲחֵרִים וּמִזְּרִימַת הָאֲוִויר בֵּינֵיהֶם.
מְאַפְיְינֵי אוֹפֶן הַחִיתּוּךְ
מִידַּת הַקִּירְבָה וְסוֹנוֹרֶנְטִיּוּת
מִידַּת הַקִּירְבָה הִיא מִידַּת הַחֲסִימָה שֶׁל מַעֲבַר הָאֲוִויר בֵּין אֵיבְרֵי הַהֲגִיָּיה בַּחֲלַל הַפֶּה וְהַגָּרוֹן, כַּאֲשֶׁר יֵשׁ סְתִימָה מְלֵאָה שֶׁל חֲלַל הַפֶּה אוֹ הַגָּרוֹן, הַחֲסִימָה שֶׁל מַעֲבַר הָאֲוִויר הִיא מוּחְלֶטֶת וּמַקְסִימָלִית, אַךְ כַּאֲשֶׁר יֵשׁ סְתִימָה חֶלְקִית שֶׁל מַעֲבַר הָאֲוִויר, הַחֲסִימָה שֶׁל מַעֲבַר הָאֲוִויר הִיא קְטַנָּה וּמִינִימָלִית.
עִם זֹאת, מִידַּת הַסּוֹנוֹרֶנְטִיּוּת הִיא עוֹצְמַת הַקּוֹל הַנִּקְבַּעַת עַל פִּי מִידַּת הַתְּנוּדָה שֶׁל מֵיתְרֵי הַקּוֹל, כַּאֲשֶׁר עוֹצְמַת הַקּוֹל נְמוּכָה, מֵיתְרֵי הַקּוֹל רְפוּיִים וְאֵינָם רוֹטְטִים, אַךְ כַּאֲשֶׁר עוֹצְמַת הַקּוֹל גְּבוֹהָה, מֵיתְרֵי הַקּוֹל רוֹטְטִים.
סוֹתֵם הוּא עִיצּוּר שֶׁבְּמַהֲלַךְ הֲפָקָתוֹ נוֹצֶרֶת חֲסִימָה מוּחְלֶטֶת וּמַקְסִימָלִית שֶׁל מַעֲבַר הָאֲוִויר בֵּין אֵיבְרֵי הַהֲגִיָּיה שֶׁמִּמֶּנָּה נוֹבֵעַ עִיקַּר צְלִיל הָעִיצּוּר וְלָכֵן מֵיתְרֵי הַקּוֹל רְפוּיִים וְאֵינָם רוֹטְטִים בְּמַהֲלַךְ הֲגִיָּיתוֹ, מִידַּת הַסּוֹנוֹרֶנְטִיּוּת שֶׁלּוֹ נְמוּכָה וְהוּא גַּם בִּלְתִּי סוֹנוֹרֶנְטִי.
בִּלְתִּי סוֹתֵם הוּא עִיצּוּר שֶׁבְּמַהֲלַךְ הֲפָקָתוֹ נוֹצֶרֶת חֲסִימָה קְטַנָּה וּמִינִימָלִית שֶׁל מַעֲבַר אֲוִויר בֵּין אֵיבְרֵי הַהֲגִיָּיה וְלָכֵן עִיקַּר צְלִיל הָעִיצּוּר נוֹבֵעַ מֵרְטִיטַת מֵיתְרֵי הַקּוֹל, מִידַּת הַסּוֹנוֹרֶנְטִיּוּת שֶׁלּוֹ גְּבוֹהָה וְהוּא גַּם סוֹנוֹרֶנְטִי.
מִידַּת הַסּוֹנוֹרֶנְטִיּוּת מוּשְׁפַּעַת מִמִּידַּת הַקִּירְבָה, כַּאֲשֶׁר הַחֲסִימָה שֶׁל מַעֲבַר הָאֲוִויר קְטַנָּה מֵיתְרֵי הַקּוֹל רוֹטְטִים וְכַאֲשֶׁר הַחֲסִימָה שֶׁל מַעֲבַר הָאֲוִויר הִיא מוּחְלֶטֶת מֵיתְרֵי הַקּוֹל רְפוּיִים וְלָכֵן עִיצּוּר סוֹתֵם הוּא עִיצּוּר בִּלְתִּי סוֹנוֹרֶנְטִי וְעִיצּוּר בִּלְתִּי סוֹתֵם הוּא עִיצּוּר סוֹנוֹרֶנְטִי.
צִדִּיּוּת
צִדִּיּוּת הִיא הָאֵזוֹר בַּחֲלַל הַפֶּה בּוֹ עוֹבֵר זֶרֶם הָאֲוִויר יַחֲסִית לְמַצַּב הַלָּשׁוֹן, הַצִּדִּיּוּת יוֹצֶרֶת הַבְחָנָה בֵּין שְׁנֵי מַצָּבִים בָּהֶם הַלָּשׁוֹן עֲשׂוּיָה לְהִמָּצֵא:
מֶרְכָּזִיִּים הֵם עִיצּוּרִים שֶׁבְּמַהֲלַךְ הֲפָקָתָם זֶרֶם הָאֲוִויר עוֹבֵר מֵעַל מֶרְכַּז הַלָּשׁוֹן. רוֹב הָעִיצּוּרִים הֵם מֶרְכָּזִיִּים.
צִדִּיִּים הֵם עִיצּוּרִים שֶׁבְּמַהֲלַךְ הֲפָקָתָם צִדֵּי הַלָּשׁוֹן יוֹרְדִים וּמְאַפְשְׁרִים אֶת מַעֲבַר זֶרֶם הָאֲוִויר מִצִּדֵּי הַלָּשׁוֹן.
כַּאֲשֶׁר לֹא מְצוּיָּן לְיַד שֵׁם הָעִיצּוּר שֶׁהוּא צִדִּי, סִימָן שֶׁהוּא מֶרְכָּזִי.
מִיּוּן אוֹפְנֵי הַחִיתּוּךְ
עִיצּוּרִים רֵיאָתִיִּים
עִיצּוּרִים רֵיאָתִיִּים הֵם עִיצּוּרִים שֶׁמּוּפָקִים בְּאֶמְצָעוּת הַמַּנְגָּנוֹן הָרֵיאָתִי מְחוּצָּן שֶׁבּוֹ זְרִימַת הָאֲוִויר הַחוּצָה מְווּסֶּתֶת בְּאֶמְצָעוּת תְּנוּעַת הַסַּרְעֶפֶת הָעוֹלָה לְמַעְלָה, יוֹצֶרֶת לַחַץ חִיּוּבִי בְּבֵית הֶחָזֶה, מְכַוֶּוצֶת אֶת הָרֵיאוֹת וְגוֹרֶמֶת לְשִׁחְרוּר הָאֲוִויר מִתּוֹכָן הַחוּצָה דֶּרֶךְ חֲלַל הַפֶּה אוֹ הָאַף. עִיצּוּרִים אֵלֶּה הֵם הַנְּפוֹצִים בְּיוֹתֵר וְיֵשׁ לָהֶם אוֹפְנֵי חִיתּוּךְ רַבִּים וְשׁוֹנִים שֶׁחֶלְקָם סוֹתְמִים וְחֶלְקָם בִּלְתִּי סוֹתְמִים.
עִיצּוּרִים בִּלְתִּי סוֹתְמִים
עִיצּוּרִים בִּלְתִּי סוֹתְמִים הֵם עִיצּוּרִים שֶׁבְּמַהֲלַךְ הֲפָקָתָם זְרִימַת הָאֲוִויר אֵינָהּ נֶחְסֶמֶת וְנוֹצָר מִרְוָוח מַסְפִּיק לְמַעֲבַר הָאֲוִויר בֵּין אֵיבְרֵי הַהֲגִיָּיה. כׇּל הָעִיצּוּרִים הַלָּלוּ הֵם רֵיאָתִיִּים.
עִיצּוּרִים מְקוֹרָבִים הֵם עִיצּוּרִים שֶׁבְּמַהֲלַךְ הֲפָקָתָם הָאֲוִויר זוֹרֵם דֶּרֶךְ מִרְוָוח שֶׁאֵינוֹ מֵבִיא לִיצִירַת צְלִיל חִיכּוּךְ. בַּהֲגִיַּית הָעִיצּוּרִים הַמְּקוֹרָבִים מֵיתְרֵי הַקּוֹל רוֹטְטִים בְּעוֹצְמָה רַבָּה, וְכִמְעַט שֶׁלֹּא נִשְׁמָע "רַעַשׁ אָקוּסְטִי". כׇּל הָעִיצּוּרִים הַמְּקוֹרָבִים מִלְּבַד הָעִיצּוּרִים הַמְּקוֹרָבִים צִדִּיִּים הֵם מֶרְכָּזִיִּים. 
חֲצָאֵי תְּנוּעוֹת (מְקוֹרָבִים) הֵם עִיצּוּרִים שֶׁיֵּשׁ לָהֶם זִיקָּה לִתְנוּעָה מְסוּיֶּמֶת וְשֶׁבְּמַהֲלַךְ הֲפָקָתָם זֶרֶם הָאֲוִויר עוֹבֵר מֵעַל מֶרְכַּז הַלָּשׁוֹן.
עִיצּוּרִים מְקוֹרָבִים נְזִילִים (חוֹכְכִים-מְקוֹרָבִים) הֵם עִיצּוּרִים שֶׁאֵין לָהֶם זִיקָּה לִתְנוּעָה מְסוּיֶּמֶת וְשֶׁלֹּא מִשְׁתַּיְּיכִים לַחֲצָאֵי הַתְּנוּעוֹת. הֵם נֶחְתָּכִים בְּאוֹפֶן מְקוֹרָב עַם חִיכּוּךְ מִזְעָרִי מוּרְגָּשׁ. 
עִיצּוּרִים מְקוֹרָבִים נְזִילִים (חוֹכְכִים-מְקוֹרָבִים) צִדִּיִּים הֵם עִיצּוּרִים שֶׁבְּמַהֲלַךְ הֲפָקָתָם זֶרֶם הָאֲוִויר עוֹבֵר מִצַּד אֶחָד אוֹ מִשְּׁנֵי צִדֵּי הַלָּשׁוֹן. גַּם הֵם נֶחְתָּכִים בְּאוֹפֶן מְקוֹרָב עִם חִיכּוּךְ מִזְעָרִי מוּרְגָּשׁ.
עִיצּוּרִים חוֹכְכִים הֵם עִיצּוּרִים שֶׁבְּמַהֲלַךְ הֲפָקָתָם הָאֲוִויר זוֹרֵם דֶּרֶךְ מִרְוָוח צַר יַחֲסִית הַמִּשְׁתַּחְרֵר עַם צְלִיל שֶׁל חִיכּוּךְ. מִבְּחִינָה אָקוּסטִית, נוֹצָר "רַעַשׁ" לְפֶרֶק זְמַן אָרוֹךְ. כׇּל הָעִיצּוּרִים הַחוֹכְכִים מִלְּבַד הָעִיצּוּרִים הַחוֹכְכִים צִדִּיִּים הֵם מֶרְכָּזִיִּים.
עִיצּוּרִים חוֹכְכִים שׁוֹרְקִים הֵם עִיצּוּרִים הַמּוּפָקִים בְּאֵזוֹר הַמַּכְתֵּשׁ וּבָתַר הַמַּכְתֵּשׁ כָּךְ שֶׁקִּדְמַת הַלָּשׁוֹן מְתַעֶלֶת אֶת זֶרֶם הָאֲוִויר הָעוֹבֵר מֵעַל מֶרְכָּזָהּ אֶל הַשִּׁינַּיִים הָעֶלְיוֹנוֹת.
עִיצּוּרִים חוֹכְכִים בִּלְתִּי-שׁוֹרְקִים הֵם עִיצּוּרִים שֶׁאֵינָם מוּפָקִים בְּאֵזוֹר הַמַּכְתֵּשׁ וּבָתַר הַמַּכְתֵּשׁ וְשֶׁלֹּא מִשְׁתַּיְּיכִים לָעִיצּוּרִים הַשּׁוֹרְקִים.
עִיצּוּרִים חוֹכְכִים בִּלְתִּי-שׁוֹרְקִים צִדִּיִּים הֵם עִיצּוּרִים שֶׁבְּמַהֲלַךְ הֲפָקָתָם זֶרֶם הָאֲוִויר עוֹבֵר מִצַּד אֶחָד אוֹ מִשְּׁנֵי צִדֵּי הַלָּשׁוֹן.
עִיצּוּרִים מְחוּכָּכִים הֵם עִיצּוּרִים שֶׁבְּמַהֲלַךְ הֲפָקָתָם זְרִימַת הָאֲוִויר נֶחְסֶמֶת בְּאוֹפֶן מוּחְלָט וּמִשְׁתַּחְרֶרֶת לְאוֹרֶךְ זְמַן תּוֹךְ יְצִירַת צְלִיל חִיכּוּךְ. מִבְּחִינָה אָקוּסטִית, נוֹצָר "רַעַשׁ" חָזָק לְפֶרֶק זְמַן קָצָר, וּמִיָּד לְאַחֲרָיו "רַעַשׁ" חַלָּשׁ יוֹתֵר לְפֶרֶק זְמַן אָרוֹךְ יוֹתֵר. כׇּל הָעִיצּוּרִים הַמְּחוּכָּכִים מִלְּבַד הָעִיצּוּרִים הַמְּחוּכָּכִים צִדִּיִּים הֵם מֶרְכָּזִיִּים. 
עִיצּוּרִים מְחוּכָּכִים שׁוֹרְקִים הֵם עִיצּוּרִים הַמּוּפָקִים בְּאֵזוֹר הַמַּכְתֵּשׁ וּבָתַר הַמַּכְתֵּשׁ כָּךְ שֶׁקִּדְמַת הַלָּשׁוֹן מְתַעֶלֶת אֶת זֶרֶם הָאֲוִויר הָעוֹבֵר מֵעַל מֶרְכָּזָהּ אֶל הַשִּׁינַּיִים הָעֶלְיוֹנוֹת.

אַנְדַּרְטַת הַקִּיבּוּצִים
אַנְדַּרְטַת הַקִּיבּוּצִים לְזֵכֶר חַבְרֵי הַקִּיבּוּצִים שֶׁנָּפְלוּ בְּמַעַרְכוֹת יִשְׂרָאֵל, נִמְצֵאת בְּיַעַר הַקִּיבּוּצִים סָמוּךְ לְמִשְׁמַר הָעֵמֶק.
הֲקָמָה
הָרַעְיוֹן לַהֲקָמַת הָאַנְדַּרְטָה נִתְקַל בְּהִתְנַגְּדוּיוֹת רַבּוֹת. הַדִּילֵמָה הָעִיקָּרִית הָיְיתָה בְּעֶצֶם נְחִיצוּת הֲקָמָתָהּ הֱיוֹת שֶׁבִּמְקוֹמוֹת רַבִּים בָּאָרֶץ יֵשׁ אַנְדַּרְטָאוֹת לְחַבְרֵי הַקִּיבּוּצִים שֶׁנָּפְלוּ בְּמַעַרְכוֹת יִשְׂרָאֵל. לְבַסּוֹף נִתְקַבְּלָה הַהַחְלָטָה וְנִבְחַר הַמָּקוֹם - גִּבְעָה נִישָּׂאָה בְּיַעַר מִשְׁמַר הָעֵמֶק. הַמָּקוֹם נִבְחַר בִּגְלַל הֱיוֹתוֹ צוֹפֶה עַל עֵמֶק יִזְרְעֶאל, עֶרֶשׂ הַהִתְיַישְּׁבוּת, וְקִרְבָתוֹ לְג'וֹעָרָה וְלִמְעָרַת הַפַּלְמָ"ח בְּקִיבּוּץ מִשְׁמַר הָעֵמֶק, דָּבָר הַמְּאַפְשֵׁר לִיצוֹר אֶת הַמְּשׁוּלָּשׁ הַוִּוירְטוּאָלִי שֶׁל מוֹרֶשֶׁת הַהֲגָנָה.
מִשְׂרַד הַבִּיטָּחוֹן הִתְנָה אֶת אִישּׁוּרוֹ וּתְמִיכָתוֹ בִּתְנַאי שֶׁלֹּא יִהְיֶה צִיּוּן שְׁמוֹת הַחֲלָלִים בָּאֲתָר. הוֹעֳלָה רַעְיוֹן לִבְנוֹת עֶמְדַּת מַחְשֵׁב בַּמָּקוֹם וּבָהּ רְשִׁימַת הַחֲלָלִים. לְבַסּוֹף הוּחְלַט לַחְקוֹק בָּאֲתָר אֶת שְׁמוֹת הַנּוֹפְלִים מֵהֶחָלָל הָרִאשׁוֹן מֹשֶׁה בֶּרְסְקִי חֲבֵר קִיבּוּץ דְּגַנְיָה א עַד יָמֵינוּ, סה"כ 3050 שֵׁמוֹת.
הָאַנְדַּרְטָה תּוּכְנְנָה עַל יְדֵי הַפַּסָּל סָלוֹ שָׁאוּל חֲבֵר קִיבּוּץ עֵין גֵּב. הִיא בְּנוּיָה בְּצוּרָה שֶׁל מִגְדַּל תַּצְפִּית עָשׂוּי מִבֵּטוֹן חָשׂוּף. גֶּרֶם מַדְרֵגוֹת מַגִּיעַ לְרֹאשׁ הַמִּגְדָּל וּמִמֶּנּוּ תַּצְפִּית מַרְהִיבָה עַל קִיבּוּץ מִשְׁמַר הָעֵמֶק וְעֵמֶק יִזְרְעֶאל. עַל קִיר בֵּטוֹן בָּאֲתָר חֲקוּקוֹת מִילּוֹתָיו שֶׁל ס. יִזְהָר: נִזְכּוֹר אַחִים, בָּנִים וְרֵעִים אֲשֶׁר יָצְאוּ עִמָּנוּ בִּפְלוּגּוֹת הַלּוֹחֲמִים. אֲנַחְנוּ חָזַרְנוּ וְהֵמָּה לֹא יָשׁוּבוּ עוֹד. קֶרֶן קַיֶּימֶת לְיִשְׂרָאֵל הִכְשִׁירָה לָאַנְדַּרְטָה דֶּרֶךְ כּוּרְכָּר מְשׁוּלֶּטֶת, בְּאוֹרֶךְ 6 ק"מ מִגּ'וֹעָרָה.
בִּתְחִילַּת הַדֶּרֶךְ, מִצַּד שְׂמֹאל, אֶפְשָׁר לְהַבְחִין בִּשְׁנֵי עַמּוּדֵי אֶבֶן גְּדוּמִים – גָּבוֹהַּ וְנָמוּךְ לְזֵכֶר שְׁנֵי חַבְרֵי קִיבּוּץ עֵין הַשּׁוֹפֵט. הַגָּבוֹהַּ לְזֵכֶר שְׁלֹמֹה קוּרְץ, הַנָּמוּךְ לְזֵכֶר נֶכְדּוֹ לִיאוֹר צָ'רָן. בְּ-7 בְּאוֹקְטוֹבֶּר 1989 יָרְדוּ הַשְּׁנַיִים לַחֲקוֹר מְעָרָה בָּאֵזוֹר וְנִקְבְּרוּ חַיִּים.
הָאֲתָר נֶחֱנַךְ רִשְׁמִית בְּ-29 בְּאוֹגוּסְט 2007 בְּטֶקֶס בְּהִשְׁתַּתְּפוּת שַׂר הַבִּיטָּחוֹן אֵהוּד בָּרָק וּנְשִׂיא הַמְּדִינָה שִׁמְעוֹן פֶּרֶס.
 אוֹפֶלְיָה שְׁטְרָאל
אוֹפֶלְיָה שְׁטְרָאל (נוֹלְדָה בְּ-7 בְּסֶפְּטֶמְבֶּר 1940) הִיא שַׂחְקָנִית קוֹלְנוֹעַ וְתֵיאַטְרוֹן יִשְׂרְאֵלִית.
קַרְיֵירָה
שְׁטְרָאל נוֹלְדָה בְּרוֹמַנְיָה. בְּגִיל 15 הֵחֵלָּה לְהוֹפִיעַ בַּתֵּיאַטְרוֹן הַיְּהוּדִי בְּבוּקָרֵשְׂט וְהִשְׁתַּתְּפָה בִּסְרָטִים. לְאַחַר שֶׁעָשְׂתָה שׁוּרָה אֲרוּכָּה שֶׁל תַּפְקִידִים עָלְתָה לְיִשְׂרָאֵל בִּשְׁנַת 1960.
בִּשְׁנַת 1965 הִשְׁתַּתְּפָה לָרִאשׁוֹנָה בַּהַצָּגָה "בְּלוּז לְמִיסְטֶר צַ'ארְלִי" בְּתֵיאַטְרוֹן הַבִּימָה, וְאַחֲרֵי כֵן שִׂחֲקָה בַּהַצָּגוֹת, "נִשְׁפֵי פֶּרֶץ", "אָנָה פְרַנְק", "גַּע בִּי בְּגַעְגּוּעִים", "הַשְּׁקִיעָה", "קָארִין", "מַלְאַךְ הַשֶּׁלֶג", "חֲבֵרוֹת", "הַקּוֹסֵם מִלּוּבְּלִין", וָעוֹד. נִיהֲלָה לַהֲקַת תֵּיאַטְרוֹן שׁוּלַיִים.
בִּשְׁנַת 1971 הִשְׁתַּתְּפָה בַּסֶּרֶט "הַמַּלְאָךְ הָיָה שָׂטָן" בְּבִימּוּיוֹ שֶׁל מֹשֶׁה גֵּז. בִּשְׁנַת 1978 הִשְׁתַּתְּפָה בַּסֶּרֶט אֶסְקִימוֹ לִימוֹן בְּתַפְקִיד "סְטֵלָה הַמְּגַמֶּרֶת", תַּפְקִיד עָלָיו קִיבְּלָה אֶת פְּרַס כִּינּוֹר דָּוִד לִשְׁנַת 1979. לְהַעֲרָכָתָהּ, תַּפְקִידָהּ כִּ"סְטֵלָה הַמְּגַמֶּרֶת" פָּגַע קָשׁוֹת בְּקַרְיֵירַת הַמִּשְׂחָק שֶׁלָּהּ.[1] לַמְרוֹת זֹאת, הִיא לֹא מִתְחָרֶטֶת עַל מִשְׂחָקָהּ בְּסֶרֶט זֶה: "לָמָּה שֶׁאֶתְחָרֵט? 'אֶסְקִימוֹ לִימוֹן' זֶה סֶרֶט שֶׁאֲנָשִׁים אָהֲבוּ. הוּא מָכַר שְׁלוֹשָׁה מִילְיוֹן כַּרְטִיס, וְזֶה כְּשֶׁהָיוּ כָּאן אַרְבָּעָה מִילְיוֹן אֲנָשִׁים. אָהֲבוּ אוֹתוֹ וְאָהֲבוּ גַּם אֶת הַדְּמוּת שֶׁלִּי, הַקָּהָל מֵת עַל סְטֵלָה הַמְּגַמֶּרֶת. עַד הַיּוֹם אֲנִי יְכוֹלָה לְקַבֵּל הֲנָחוֹת אִם רַק אָבוֹא לַחֲנוּיוֹת וְאַגִּיד שֶׁאֲנִי סְטֵלָה הַמְּגַמֶּרֶת, אֲבָל אֲנִי אִשָּׁה תַּרְבּוּתִית מִכְּדֵי לַעֲשׂוֹת דָּבָר כָּזֶה".[2]
בַּשָּׁנִים שֶׁאַחֲרֵי כֵן הִשְׁתַּתְּפָה בַּסְּרָטִים "טָעוּת בַּמִּסְפָּר" (עִם שַׁיְיקֶה אוֹפִיר), "לֹא לַעֲלוֹת יוֹתֵר" (לְצַד זְאֵב רֶוַוח) וְ"יִשְׂרְאֵלִים מַצְחִיקִים".
בַּשָּׁנִים 1983–1990 שִׂחֲקָה בְּהַצָּגוֹת יָחִיד "נָשִׁים וּזְמַנִּים קָשִׁים", "גַּן עֵדֶן" וְ"הַפִּילֶגֶשׁ" וְכֵן הִשְׁתַּתְּפָה בַּהַצָּגוֹת "בִּדְלָתַיִים סְגוּרוֹת", "כָּלְאָדָם" (1984), "נָשִׁים בַּתֵּיאַטְרוֹן" (1986), "קַיִן הָעֲנָק", וָעוֹד.
בִּשְׁנַת 2006 הוֹפִיעָה בְּהַצָּגַת יָחִיד בְּשֵׁם "סִיפּוּרֵי אוֹפֶלְיָה", שֶׁבָּהּ תֵּיאֲרָה אֶת יַלְדוּתָהּ וְהַתַּפְקִידִים הַשּׁוֹנִים שֶׁגִּילְּמָה בַּתֵּיאַטְרוֹן וּבַקּוֹלְנוֹעַ.
בִּשְׁנַת 2011 הִשְׁתַּתְּפָה בַּסֶּרֶט הַתִּיעוּדִי "לְאָן נֶעֱלַם מֹשֶׁה גֵּז" בְּבִימּוּיוֹ שֶׁל אֲבִידָע לִיבְנִי. בְּמַהֲלַךְ הַסֶּרֶט, הַמְּתַעֵד אֶת חִיפּוּשׂוֹ שֶׁל לִיבְנִי אַחֲרֵי בַּמַּאי "הַמַּלְאָךְ הָיָה שָׂטָן" חוֹשֶׂפֶת שְׁטְרָאל אֶת הַתְּחוּשׁוֹת שֶׁלִּיוּוּ אוֹתָהּ בְּמַהֲלַךְ צִילּוּמֵי הַסֶּרֶט, וּלְאוֹרֶךְ הַשָּׁנִים שֶׁחָלְפוּ מֵאָז.
לִקְרִיאָה נוֹסֶפֶת
אוֹפֶלְיָה שְׁטְרָאל, לֵיל הַפְּנִינִים, הוֹצָאַת קֶרֶן, 1996
 א גְּרוֹיְיסֶע מְצִיאֶה
אַ גְּרוֹיְיסֶע מְצִיאֶה (מִיִּידִישׁ, לָרוֹב בְּסַרְקַזְם: מְצִיאָה גְּדוֹלָה) הִיא לַהֲקַת פַּאנְק וּגְרוּב יִשְׂרְאֵלִית שֶׁפָּעֲלָה בֵּין הַשָּׁנִים 2006–2012.
בַּלַּהֲקָה הָיוּ חֲבֵרִים שִׁישָּׁה מוּזִיקָאִים - מִתּוֹכָם אַרְבָּעָה חוֹזְרִים בִּתְשׁוּבָה וְשֵׁנִי חִילּוֹנִיִּים.
בֵּין הַשְּׁאָר, הוֹפִיעָה הַלַּהֲקָה לְצַד אַהֲרֹן רַזְאֵל, דָּוִד דֵּ'אוֹר, עָמִיר בְּנָיוּן וָעוֹד.
בְּ-2013, לְאַחַר פֵּירוּק הַלַּהֲקָה, הִתְאַחֲדוּ לְצוֹרֶךְ בִּיצּוּעַ הַשִּׁיר "וְאַהֲבַת לְרֵעֲךָ כָּמוֹךָ" בָּאַלְבּוֹם "שִׂמְחַת עוֹלָם" הַמְּאַגֵּד שִׁירִים שֶׁהִלְחִין הָרַב יִצְחָק גִּינְזְבּוּרְג. אֶת הָאַלְבּוֹם הֵפִיקוּ מוּזִיקָלִית נָדָב בָּכָר וְאוֹרֶן צוּר.
דִּיסְקוֹגְרַפְיָה
 אֵלִיָּהוּ מִזְרָחִי
רַבִּי אֵלִיָּהוּ מִזְרָחִי (הָרְאֵ"ם; 1435–1526,רפ"ו) הָיָה פּוֹסֵק הֲלָכָה, אִישׁ מַדָּע, וּמְדִינַאי יְהוּדִי-טוּרְקִי. רַבָּהּ הָרָאשִׁי שֶׁל יַהֲדוּת טוּרְקִיָּה וּמַנְהִיגָהּ, רֹאשׁ יְשִׁיבָה וְאֶחָד מִשְּׁלוֹשֶׁת פַּרְשָׁנָיו הָעִיקָּרִיִּים שֶׁל פֵּירוּשׁ רָשִׁ"י לַתּוֹרָה.
תּוֹלְדוֹת חַיָּיו
הָרְאֵ"ם נוֹלַד בְּקוּשְׁטָא בֵּין הַשָּׁנִים 1435 - 1437 (וְיֵשׁ הַגּוֹרְסִים 1450), לְמִשְׁפָּחָה רוֹמַנְיוֹטִית מִן הַבַּלְקָן. לָמַד תּוֹרָה אֵצֶל רַבִּי אֵלִיָּהוּ הַלֵּוִי בֶּן רַבִּי מֹשֶׁה הַלֵּוִי הָאַשְׁכְּנָזִי, וְאֵצֶל מָרְדֳּכַי כּוֹמְטִינוֹ לָמַד מָתֵמָטִיקָה וְאַסְטְרוֹנוֹמְיָה.
הֶעֱמִיד תַּלְמִידִים רַבִּים, בְּחוֹכְמַת הַתּוֹרָה וּבְחוֹכְמוֹת חִיצוֹנִיּוֹת (הָרְאֵ"ם שָׁלַט גַּם בַּעֲרָבִית וּבִיוָונִית וְנֶאֱמַר עָלָיו כִּי הָיָה הָרִאשׁוֹן שֶׁגִּילָּה שִׁיטָה לִמְצִיאַת שׁוֹרֶשׁ רִיבּוּעִי). בֵּין הַתַּלְמִידִים שֶׁהֶעֱמִיד הָיוּ גַּם תַּלְמִידִים קָרָאִים וְזֹאת לְמוֹרַת רוּחוֹ שֶׁל הָרַב מֹשֶׁה אֵלִיָּהוּ קַפְּשָׂאלִי, שֶׁהָיָה הַסַּמְכוּת הַתּוֹרָנִית הָעֶלְיוֹנָה בָּאִימְפֶּרְיָה הָעוֹתְ'מָאנִית בְּאוֹתָהּ עֵת. בְּכָךְ וּבְנוֹשְׂאִים נוֹסָפִים הִגִּיעַ לְעִימּוּת עַם הָרַב קַאפְּסָלִי. עִם זֹאת, עַל אַף הוֹרָאָתוֹ שֶׁל הָרַב קוֹלוֹן (המהרי"ק) שֶׁלֹּא לְצַיֵּית לָרַב קַאפְּסָלִי בְּתַפְקִידוֹ כֶּחָכָם בָּאשִׁי (תַּפְקִיד שֶׁכָּלַל גַּם סַמְכוּיוֹת לִגְבִיַּת כְּסָפִים שׁוֹנִים מֵהַיְּהוּדִים עֲבוּר הַשּׂוּלְטָן), נִמְנַע הָרְאֵ"ם בַּעֲצַת רַבּוֹ, הָרַב יְהוּדָה מִינְץ, מִלְּהַבִּיעַ עֶמְדָּה בַּסּוּגְיָה.
חָכָם בָּאשִׁי[1]
לְאַחַר מוֹת הָרַב מֹשֶׁה אֵלִיָּהוּ קַפְּשָׂאלִי בְּ-1495 מוּנָּה בִּמְקוֹמוֹ הָרְאֵ"ם לְחָכָם בָּאשִׁי, וּלְמַעֲשֶׂה הָיָה לַחֲכַם הַבָּאשִׁי הָאַחֲרוֹן. בִּיטֵּל אֶת הַחֲרָמוֹת שֶׁהוּטְּלוּ עַל לִימּוּד תּוֹרָה לְקָרָאִים וְנִימֵּק אֶת הַדָּבָר בְּכָךְ שֶׁקִּנְאַת תַּלְמִידִים תַּרְבֶּה חוֹכְמָה[2]. בְּסוֹבְלָנוּתוֹ הָיָה הָרְאֵ"ם יוֹצֵא מִן הַכְּלָל בִּזְמַנּוֹ וּמְקוֹמוֹ. אֲבָל לַמְרוֹת זֶה בִּיסֵּס אֶת אִיסּוּר הַחִיתּוּן עִם הַקָּרָאִים[3].
הָרְאֵ"ם גִּילָּה יַחַס מֵקֵל לְהַתָּרַת עֲגוּנוֹת מִטַּעֲמֵי צְנִיעוּת. הָרְאֵ"ם קָבַע בְּדַעַת מִיעוּט כִּי גַּם לְאַחַר חֵרֶם דְּרַבֵּנוּ גֵּרְשׁוֹם, מָקוֹם בּוֹ יַפְקִיד בַּעַל אֶת הַגֵּט וְהַכְּתוּבָּה בִּידֵי בֵּית הַדִּין, יִהְיֶה פָּטוּר מִתַּשְׁלוּם דְּמֵי מְזוֹנוֹת לְאִשְׁתּוֹ. הִלְכַת מִיעוּט זוֹ אוֹמְנָם לֹא אוּמְּצָה עַל יְדֵי מַרְבִּית הָרַבָּנִים, אַךְ אוּמְּצָה בְּבָתֵּי הַמִּשְׁפָּט לְעִנְיְינֵי מִשְׁפָּחָה, בְּמָקוֹם בּוֹ קַיָּים פֵּירוּד אָרוֹךְ בֵּין בְּנֵי זוּג, וְהָאִשָּׁה דּוֹחָה אֶפְשָׁרוּת חֲזָרָה לְחַיֵּי מִשְׁפָּחָה, רְאוּ לְמָשָׁל כָּאן.
הָרְאֵ"ם הִתְנַגֵּד בַּחֲרִיפוּת לְמִנְהָגִים שֶׁנִּתְפְּסוּ בְּעֵינָיו כִּגְרוּעִים וְאָמַר כִּי "מִנְהָג גָּרוּעַ שֶׁאֵינוֹ עַל פִּי הָעִיקָּר, מִנְהָג בָּטֵל הוּא אֵצֶל כׇּל אָדָם וְאֵין מְחַיְּיבִים לָלֶכֶת אַחֲרָיו אַף עַל גַּב דְּאִתְפַּשַּׁט אִסּוּרֵיהּ לְכוּלָּא מָתָא (=אַף עַל גַּב שֶׁהָאִיסּוּר פָּשַׁט בְּכׇל אוֹתָהּ הָעִיר) וְאַף עַל גַּב דְּהָוֵי גַּבֵּי אִסּוּרָא...". וְעוֹד כָּתַב "דְּאֵין כֹּחַ הַמִּנְהָג יָכוֹל לְהִתְפַּשֵּׁט עַל מִי שֶׁאֵינוֹ רוֹצֶה בְּאוֹתוֹ מִנְהָג בְּעַל כׇּרְחוֹ".
הָרְאֵ"ם הִתְפַּרְסֵם בְּיַדְעָנוּתוֹ, וְעָסַק הַרְבֵּה בְּצׇרְכֵי צִיבּוּר לַמְרוֹת חוּלְשָׁתוֹ הַגּוּפָנִית, וְעִם מִינּוּיוֹ לְחָכָם בָּאשִׁי הָיָה מִיּוֹעֲצֵי הַשּׂוּלְטָן בְּאִיזְמִיר, וְהָיָה לוֹ מָקוֹם בַּדִּיוָאן עַם הַפַּטְרִיאַרְךְ וְהַמּוּפְתִּי. בְּיָמָיו הִתְחִילָה כְּנִיסַת הֲמוֹנֵי יְהוּדִים לְטוּרְקִיָּה בְּעִקְבוֹת גֵּירוּשׁ סְפָרַד, וְהָרְאֵ"ם עָשָׂה מַאֲמַצִּים לְהָקֵל עַל הַבָּאִים אֶת כְּנִיסָתָם לָאָרֶץ הַחֲדָשָׁה.
הָרְאֵ"ם נִפְטַר בְּגִיל מוּפְלָג בְּקוּשְׁטָא בִּשְׁנַת 1526. חֲתָנוֹ, רַבִּי מְשׁוּלָּם מִוּוֹלְטִירָה הֵבִיא לְבִיטּוּל הָרַבָּנוּת הָרָאשִׁית בְּתוֹגַרְמָה (טוּרְקִיָּה) לְאַחַר פְּטִירָתוֹ. הָרִאשׁוֹן לְצִיּוֹן רַבִּי נִיסִּים חַיִּים מֹשֶׁה מִזְרָחִי וְיִשְׂרָאֵל מֵאִיר מִזְרָחִי מִיְּרוּשָׁלַיִם הֵם מִצֶּאֱצָאָיו שֶׁל הָרְאֵ"ם.
סְפָרָיו
הַמִּזְרָחִי - פֵּירוּשׁ עַל פרש"י שֶׁעַל הַתּוֹרָה (וִינִיצְיָא 1526, 1545) - סֵפֶר שֶׁבּוֹ הֶעֱמִיק לְפָרֵשׁ אֶת דִּבְרֵי רָשִׁ"י, לְהָגֵן עֲלֵיהֶם מֵהַשָּׂגוֹת הַמְּפָרְשִׁים הָאַחֲרוֹנִים (וּבִמְיוּחָד מֵהַשָּׂגוֹת הָרַמְבָּ"ן) אַךְ גַּם לַחֲלוֹק עָלָיו בִּמְקוֹמוֹת רַבִּים. הַסֵּפֶר נֶחְשָׁב לְאַחַד הַפֵּירוּשִׁים הַמֶּרְכָּזִיִּים לְרָשִׁ"י. הַסֵּפֶר מֵבִיא לְמַעֲשֶׂה אֶת הַמְּקוֹרוֹת מֵהַתַּלְמוּד וּמֵהַמִּדְרָשׁ עֲלֵיהֶם בִּיסֵּס הרש"י אֶת פֵּירוּשׁוֹ. תַּמְצִיתוֹ מוֹפִיעָה בַּסֵּפֶר "שִׂפְתֵי חֲכָמִים" (עִם פֵּירוּשׁוֹ שֶׁל הַמָּהָרָ"ל מִפְּרָאג).
מְלֶאכֶת הַמִּסְפָּר - עַל חוֹכְמַת הַמִּסְפָּר הַחֶשְׁבּוֹן וְהַתִּשְׁבּוֹרֶת וְהַהַנְדָּסָה (קוּשְׁטָא 1534) - מַטְּרַת סֵפֶר זֶה הִיא לְלַמֵּד כׇּל אָדָם אֶת יְסוֹדוֹת הַחֶשְׁבּוֹן וְאֶת אַרְבַּע פְּעוּלּוֹת הַחֶשְׁבּוֹן: קִיבּוּץ (חִיבּוּר), חִיסּוּר, הַכָּאָה (כֶּפֶל) וְחִילּוּק, בְּמִסְפָּרִים שְׁלֵמִים, בִּשְׁבָרִים וּבְשִׁבְרֵי הַתְּכוּנָה. אֶת הַסֵּפֶר הוֹצִיא לָאוֹר בְּנוֹ שֶׁל הָרְאֵ"ם, רַבִּי יִשְׂרָאֵל[4], לְאַחַר פְּטִירָתוֹ. הַסֵּפֶר כָּתוּב עִבְרִית וְתוּרְגַּם לְלָטִינִית, הִתְפַּרְסֵם בַּחֲשִׁיבוּתוֹ וְשִׁימֵּשׁ כְּסֵפֶר יְסוֹד לְהוֹרָאַת הַחֶשְׁבּוֹן. בַּסֵּפֶר כָּלוּל גַּם פֶּרֶק עַל לִימּוּדֵי הַשָּׁח, רְאוּ כָּאן.
תּוֹסְפֵי סמ"ג (קוּשְׁטָא 1541) - חִידּוּשִׁים בְּסֵפֶר מִצְווֹת גָּדוֹל.
צוּרַת הָאָרֶץ - חִיבּוּר עַל צוּרַת הָאָרֶץ וּתְמוּנָתָהּ.
שְׁאֵלוֹת וּתְשׁוּבוֹת (קוּשְׁטָא 1546) - מֵאָה שְׁאֵלוֹת וּתְשׁוּבוֹת בְּשִׁבְתּוֹ עַל כֵּס הַהוֹרָאָה.
מַיִם עֲמוּקִּים (וִינִיצְיָה 1647) - שְׁאֵלוֹת וּתְשׁוּבוֹת חֵלֶק ב'.
פְּסָקִים שֶׁפּוּרְסְמוּ בְּסֵפֶר בֵּית יוֹסֵף לְרַבִּי יוֹסֵף קָארוֹ
סְפָרִים שֶׁלֹּא נִדְפְּסוּ
בֵּאוּר עַל סֵפֶר אַלְמָגִיסְטִי - סֵפֶר עַל תְּנוּעַת גַּרְמֵי הַשָּׁמַיִם - הַסֵּפֶר לֹא נִדְפַּס אַךְ קַיָּים כְּתַב יָד בְּסֵנְט פֵּטֶרְבּוּרְג.
פֵּירוּשׁ לִיסוֹדוֹת שֶׁל אֵוּקְלִידֶס - הַסֵּפֶר אָבַד.
לִקְרִיאָה נוֹסֶפֶת
א' עוֹבַדְיָה, כְּתָבִים נִבְחָרִים כֶּרֶךְ א, יְרוּשָׁלַיִם תש"ב, עמ' 98-63.
מ' פִילִיפּ, חוּמַּשׁ הָרְאֵ"ם, סֵפֶר בְּרֵאשִׁית, פֶּתַח תִּקְוָוה תשנ"ד, מָבוֹא עמ' ה-כג.
 אוּזְבֵּקִים
אוּזְבֵּקִים הֵם קְבוּצָה אֶתְנִית בְּמֶרְכַּז אַסְיָה. רוּבָּהּ שֶׁל קְבוּצָה זוֹ נִמְצָא בְּאוּזְבֵּקִיסְטָן, אֲבָל נִיתָּן לִמְצוֹא אוּזְבֵּקִים גַּם בִּמְדִינוֹת אֲחֵרוֹת כְּגוֹן טָגִ'יקִיסְטָן וְאַפְגָּנִיסְטָן. הָאוּכְלוּסִיָּיה כּוּלָּהּ מוֹנָה בֵּין 28 לִ-30 מִילְיוֹן אִישׁ.
מְקוֹר הַשֵּׁם
מְקוֹר הַשֵּׁם 'אוּזְבֵּקִים' לֹא יָדוּעַ, אֲבָל יֶשְׁנָן הַשְׁעָרוֹת בְּנוֹגֵעַ לְכָךְ. הַשְׁעָרָה אַחַת אוֹמֶרֶת שֶׁהֵם נִקְרְאוּ עַל שֵׁם אוּזְבֶּג חָאן.[1] הַשְׁעָרָה נוֹסֶפֶת הִיא שֶׁמַּשְׁמָעוּת הַמִּילָּה הִיא "עַצְמָאוּת" בְּשָׂפָה טוּרְקִית עַתִּיקָה.
שָׂפָה
הַשָּׂפָה הָאוּזְבֵּקִית הִיא שָׂפָה טוּרְקִית. אוּזְבֵּקִית מוֹדֶרְנִית נִכְתֶּבֶת ב כמה סוּגִים שֶׁל אוֹתִיּוֹת: לָטִינִיּוֹת, קִירִילִיּוֹת וַעֲרָבִיּוֹת. לְאַחַר שֶׁאוּזְבֵּקִיסְטָן קִיבְּלָה עַצְמָאוּת מִבְּרִית הַמּוֹעָצוֹת הֶחְלִיטוּ לְהוֹרִיד אֶת הָאוֹפְּצִיָּה לִכְתּוֹב בְּאַלְפָאבֵּיתִית קִירִילִי.
דָּת
מַרְבִּית הָאוּזְבֵּקִים הֵם מוּסְלְמִים סוּנִים מֵהָאַסְכּוֹלָה הַחָנָפִית[2], אַךְ קַיָּימִים כָּאֵלֶּה שֶׁלֹּא מְשׁוּיָּכִים לָאִסְלָאם. הָאוּזְבֵּקִים בִּמְדִינוֹת שֶׁלֹּא מְעוֹדֶדֶת קִיצוֹנִיּוּת אִסְלָאמִית (כְּמוֹ אוּזְבֵּקִיסְטָן וְרוּסְיָה) לֹא קִיצוֹנִיִּים כְּלָל. לְעוּמַּת זֹאת, מוּסְלְמִים בִּמְדִינוֹת אִסְלָאמִיּוֹת קִיצוֹנִיּוֹת (כְּמוֹ אַפְגָּאנִיסְטָן) יוֹתֵר קִיצוֹנִיִּים וְשׁוֹמְרִים בִּקְפִידוּת עַל הַדָּת. בִּתְקוּפַת בְּרִית הַמּוֹעָצוֹת הַדָּת לֹא נִשְׁאֲרָה חֲשׁוּבָה וְהוּזְנְחָה, אַךְ כַּאֲשֶׁר בְּרִית הַמּוֹעָצוֹת הִתְפָּרְקָה הִתְרַחֲשָׁה תְּחִיָּיה לַדָּת. תַּהֲלִיךְ זֶה נִמְשַׁךְ עַד לַמַּצָּב כַּיּוֹם, שֶׁבּוֹ 96.3% מֵאוּכְלוּסִיַּית אוּזְבֵּקִיסְטָן הֵם אוּזְבֵּקִים מוּסְלְמִים.[3]
 אוֹרֶן מַיְילִי
אוֹרֶן מַיְילִי (מְכוּנֶּה אוֹרֶן הַקָּטָן; נוֹלַד בִּשְׁנַת 1978) הוּא קוֹסֵם וְאׇמַּן טֵלֶפַּתְיָה יִשְׂרְאֵלִי, שֶׁהִתְפַּרְסֵם בְּיִשְׂרָאֵל בִּשְׁנוֹת הַ-80 וּבִתְחִילַּת שְׁנוֹת הַ-90 שֶׁל הַמֵּאָה הָ-20.
בִּיּוֹגְרַפְיָה
מַיְילִי נוֹלַד וְגָדַל בִּשְׁכוּנַת פַּרְדֵּס כַּץ בִּבְנֵי בְּרָק. הוּא הֵחֵל אֶת דַּרְכּוֹ בַּפִּינָּה שֶׁהוּקְצְתָה לוֹ אֵצֶל הַבַּדְרָן מֵאִיר רֶמֶז. אֶת הוֹפָעוֹתָיו הָרִאשׁוֹנוֹת הֵחֵל בְּגִיל פָּחוֹת מֵעֶשֶׂר, וּמִכָּאן כִּינּוּיוֹ "אוֹרֶן הַקָּטָן". בִּשְׁנוֹת הַ-80 וּבִתְחִילַּת שְׁנוֹת הַ-90 שֶׁל הַמֵּאָה הָ-20 הוֹפִיעַ בְּמוֹפָעִים וּבְתוֹכְנִיּוֹת טֵלֵוִויזְיָה יַחַד עִם אָבִיו, רוֹנִי מַיְילִי.
בְּאֶמְצַע שְׁנוֹת הַתִּשְׁעִים עָזַב אֶת עוֹלַם הַבִּידּוּר וְהֵחֵל לְסַיֵּיעַ לְאָבִיו בְּעֵסֶק לְיִיצּוּר שִׁלְטֵי פִּרְסוּם. כַּיּוֹם הוּא מוֹפִיעַ לִפְנֵי וַעֲדֵי עוֹבְדִים בְּהַנְחָיַית אָבִיו.
בְּ-2006 זוּמַּן לַמִּשְׁטָרָה, בִּגְלַל בִּלְבּוּל שֶׁהָיָה בֵּינוֹ לְבֵין אוֹרֶן זָרִיף, בַּחֲשָׁד שֶׁהוֹנָה אֲנָשִׁים שֶׁבָּאוּ לְטִיפּוּל אֶצְלוֹ.[1]
בְּ-2013 הִשְׁתַּתְּפוּ בְּתוֹכְנִית "הָאָח הַגָּדוֹל" אָבִיו שֶׁל אוֹרֶן, רוֹנִי מַיְילִי, וְאָחִיו גִּילִי, שֶׁמּוֹפִיעִים לְעִיתִּים לְצִדּוֹ בְּהוֹפָעוֹתָיו.[2]
מַיְילִי נָשׂוּי וְאָב לִשְׁלוֹשָׁה יְלָדִים.
 אָלֵכְּס הִיגִינְס
אָלֶכְּסַנְדֶּר גּוֹרְדּוֹן "אָלֵכְּס" הִיגִינְס (18 בְּמֵרְץ 1949 - 24 בְּיוּלִי 2010), יָדוּעַ גַּם בַּכִּינּוּי הוֹרִיקָן הִיגִנְס, הָיָה שַׂחְקַן סְנוּקֶר מִקְצוֹעָנִי צְפוֹן אִירִי. הִיגִינְס זָכָה פַּעֲמַיִים בְּאַלִּיפוּת הָעוֹלָם בִּסְנוּקֶר בַּשָּׁנִים 1972 וְ-1982, וְהָיָה פַּעֲמַיִים סְגַן אַלּוּף הָעוֹלָם בִּסְנוּקֶר בַּשָּׁנִים 1976 ו-1980. בְּנוֹסָף, הִיגִינְס זָכָה בְּאַלִּיפוּת הָעוֹלָם בִּסְנוּקֶר לְזוּגוֹת, יַחַד עִם גֶ'יְמִי וַיְיט הָאַנְגְּלִי בִּשְׁנַת 1984, וְכֵן זָכָה שָׁלוֹשׁ פְּעָמִים בִּרְצִיפוּת בִּגְבִיעַ הָעוֹלָם בִּסְנוּקֶר לְנִבְחָרוֹת, עִם נִבְחֶרֶת אִירְלַנְד בַּשָּׁנִים 1985, 1986 וְ-1987. בְּשִׂיאוֹ דּוֹרַג הִיגִינְס בַּמָּקוֹם הַשֵּׁנִי בָּעוֹלָם בְּדֵירוּג שַׂחְקָנֵי הַסְּנוּקֶר הָעוֹלָמִי. זָכָה לַכִּינּוּי "הוֹרִיקָן" בִּזְכוּת סִגְנוֹן מִשְׂחֲקוֹ הַמָּהִיר.
קַרְיֵירָה
הִיגִינְס הֵחֵל לְשַׂחֵק סְנוּקֶר מִקְצוֹעָנִי בְּעוֹנַת 1971/1972, בְּגִיל 22. הוּא זָכָה בְּאַלִּיפוּת הָעוֹלָם בִּסְנוּקֶר כְּבָר בְּנִיסְיוֹנוֹ הָרִאשׁוֹן בִּשְׁנַת 1972, בְּגִיל 22, לְאַחַר שֶׁנִּיצֵּחַ בַּגְּמָר אֶת ג'וֹן סְפֶּנְסֶר הָאַנְגְּלִי בַּתּוֹצָאָה 32–37, וּבְכָךְ נִהְיָה לְאַלּוּף הָעוֹלָם בִּסְנוּקֶר הַצָּעִיר בְּיוֹתֵר אֵי פַּעַם (שִׂיא שֶׁנִּשְׁבַּר כַּעֲבוֹר 18 שָׁנִים עַל יְדֵי סְטִיבֶן הֶנְדְרִי כְּשֶׁזָּכָה בְּאַלִּיפוּת הָעוֹלָם בְּגִיל 21). לְאַחַר מִכֵּן הִגִּיעַ הִיגִינְס לִגְמַר אַלִּיפוּת הָעוֹלָם בַּשָּׁנִים 1976 וְ-1980, אַךְ הִפְסִיד. בִּשְׁנַת 1982, זָכָה הִיגִינְס בַּפַּעַם הַשְּׁנִיָּיה בְּאַלִּיפוּת הָעוֹלָם, לְאַחַר שֶׁנִּיצֵּחַ בַּגְּמָר אֶת רֵיי רִירְדוֹן הַוֶּלְשִׁי בַּתּוֹצָאָה 15–18, וְאִם לֹא הָיָה נִקְנַס בְּהוֹרָדַת נְקוּדּוֹת דֵּירוּג עֵקֶב עֲבֵירוֹת מִשְׁמַעַת, הָיָה מְדוֹרָג בְּעוֹנַת 1982/1983 בַּמָּקוֹם הָרִאשׁוֹן בָּעוֹלָם.
הִיגִינְס, עִם סִגְנוֹן מִשְׂחֲקוֹ, שֶׁהָיָה מַלְהִיב, הָיָה אַחְרַאי בְּמִידָּה רַבָּה לָעֲלִיָּיה בַּפּוֹפּוּלָרִיּוּת שֶׁל עֲנַף הַסְּנוּקֶר בְּסוֹף שְׁנוֹת הַשִּׁבְעִים. הוּא הָיָה שַׂחְקָן מָהִיר מְאוֹד, וּבְרַאֲיוֹנוֹת הוּא אַף הִסְבִּיר שֶׁהַסִּיבָּה לְכָךְ הָיְיתָה פָּשׁוּט חוֹסֶר סַבְלָנוּת. רַבִּים מַחֲשִׁיבִים אוֹתוֹ לְאֶחָד מִשַּׂחְקָנֵי הַסְּנוּקֶר הַטּוֹבִים בְּיוֹתֵר בְּכׇל הַזְּמַנִּים.
עִם זֹאת, נוֹדַע גַּם בְּאוֹפִי וּמֶזֶג בְּעָיָיתִיִּים. עַל פִּי הַעֲרָכוֹת, כַּ-50 תְּלוּנוֹת רִשְׁמִיּוֹת הוּגְּשׁוּ נֶגֶד אָלֵכְּס הִיגִינְס עַל יְדֵי הִתְאַחֲדוּת הַסְּנוּקֶר הָעוֹלָמִית - יוֹתֵר מִכׇּל שַׂחְקַן סְנוּקֶר אַחֵר בַּהִיסְטוֹרְיָה שֶׁל הַמִּשְׂחָק.[1]
בְּמַהֲלַךְ הַקַּרְיֵירָה, זָכָה הִיגִינְס בִּ-30 תְּאָרִים מִקְצוֹעָנִיִּים (26 תּוֹאֳרֵי יְחִידִים וְ-4 תְּאָרִים קְבוּצָתִיִּים), בָּהֶם אַלִּיפוּת בְּרִיטַנְיָה בִּשְׁנַת 1983, טוּרְנִיר הַמַּאסְטֵרְס בַּשָּׁנִים 1978 וְ-1981, וְהַמַּאסְטֵרְס הָאִירִי בִּשְׁנַת 1989, בּוֹ נִיצֵּחַ בְּגִיל 40 אֶת סְטִיבֶן הֶנְדְרִי הַצָּעִיר.
 אֶלְדָּד פֶּלֶד
אֶלְדָּד פֶּלֶד הוּא קְצִין צַהַ"ל בְּמִילּוּאִים בְּדַרְגַּת אַלּוּף מִשְׁנֶה. 
שֵׁירוּת צְבָאִי
עִם גִּיּוּסוֹ לְצַהַ"ל שׁוּבַּץ פֶּלֶד בַּחֲטִיבַת גִּבְעָתִי וְעָבַר מַסְלוּל הַכְשָׁרָה כְּלוֹחֵם, קוּרְס מַ"כִּים חִי"ר וְקוּרְס קְצִינֵי חִי"ר. עַם סִיּוּם הַקּוּרְס שִׁימֵּשׁ כִּמְפַקֵּד מַחְלָקָה וּפְלוּגָּה בַּחֲטִיבָה. בַּהֶמְשֵׁךְ שִׁימֵּשׁ כִּמְפַקֵּד פלחה"ן גִּבְעָתִי וְכִמְפַקֵּד סַיֶּירֶת גִּבְעָתִי. בַּשָּׁנִים 1999–2001 שִׁימֵּשׁ כִּמְפַקֵּד גְּדוּד שָׁקֵד וְלָקַח חֵלֶק בַּפְּעִילוּת הַמִּבְצָעִית כְּנֶגֶד הַטֵּרוֹר הַפָלַסְטִינִי בִּרְצוּעַת עַזָּה וּבַלְּחִימָה בִּדְרוֹם לְבָנוֹן. לְאַחַר מִכֵּן שִׁימֵּשׁ כִּמְפַקֵּד בא"ח גִּבְעָתִי[1] וְכִסְגַן מְפַקֵּד הַחֲטִיבָה[2]. בַּשָּׁנִים 2006–2009 שִׁימֵּשׁ כִּמְפַקֵּד מַעֲרַךְ מַגָּל[3]. לְאַחַר שִׁחְרוּרוֹ פָּנָה לַעֲבוֹד כְּעוֹרֵךְ דִּין[4].
אַיְירִין לֵדֶרְמָן
אַיְירִין לֵדֶרְמָן הִיא אַתְלֵטִית יְהוּדִיָּיה-בְּרִיטִית בֶּעָבָר, אַלּוּפָה וְשִׂיאָנִית הַמַּכַּבִּיָּה בַּעֲלַת 6 מֵדַלְיוֹת, מֵהֵן 4 מֵדַלְיוֹת זָהָב (הַשְּׁלִישִׁית 1950 וְהָרְבִיעִית 1953) בְּרִיצַת 100 מֶטֶר, בְּרִיצַת 200 מֶטֶר וּבִקְפִיצָה לָרוֹחַק.
בִּיּוֹגְרַפְיָה
לֵדֶרְמָן גָּדְלָה בְּלוֹנְדוֹן.
אַלּוּפָה וְשִׂיאָנִית הַמַּכַּבִּיָּה
בְּתַחֲרוּיוֹת הָאַתְלֵטִיקָה שֶׁל הַמַּכַּבִּיָּה הַשְּׁלִישִׁית שֶׁנֶּעֶרְכוּ בִּשְׁנַת 1950 זָכְתָה בְּ-3 מֵדַלְיוֹת, אַחַת מִכׇּל סוּג. מֵדַלְיַית זָהָב, בְּרִיצַת 100 מֶטֶר בְּשִׂיא מַכַּבִּיָּה חָדָשׁ שֶׁל 12.8 שְׁנִיּוֹת, לִפְנֵי עֶדְנָה פְרִידְמָן מִדְּרוֹם אַפְרִיקָה (12.8 שְׁנִיּוֹת) וְלֵאָה הוֹרְבִיץ מִיִּשְׂרָאֵל (13.3 שְׁנִיּוֹת), מֵדַלְיַית כֶּסֶף, בְּרִיצַת 60 מֶטֶר בְּתוֹצָאָה שֶׁל 8.3 שְׁנִיּוֹת, אַחֲרֵי עֶדְנָה פְרִידְמָן 8.2 שְׁנִיּוֹת וְלִפְנֵי וַן דֶר שְׁטָאם מֵהוֹלַנְד (8.3 שְׁנִיּוֹת) וּמֵדַלְיַית אָרָד, בְּרִיצַת 80 מֶטֶר מְשׂוּכוֹת בְּתוֹצָאָה שֶׁל 15.6 שְׁנִיּוֹת, אַחֲרֵי רוּת אַיָּילוֹן (13.8 שְׁנִיּוֹת) וְשָׂרָה מִיזְרָחִי (15.4 שְׁנִיּוֹת), שְׁתֵּיהֶן מִיִּשְׂרָאֵל.[1][2]
כּוֹכֶבֶת הַמַּכַּבִּיָּה הָרְבִיעִית
בְּתַחֲרוּיוֹת הָאַתְלֵטִיקָה שֶׁל הַמַּכַּבִּיָּה הָרְבִיעִית שֶׁנֶּעֶרְכוּ בִּשְׁנַת 1953 זָכְתָה פְרִידְמָן בְּ-3 מֵדַלְיוֹת זָהָב יוֹתֵר מִכׇּל סְפּוֹרְטַאי אַחֵר, בְּרִיצַת 100 מֶטֶר בְּתוֹצָאָה שֶׁל 12.7 שְׁנִיּוֹת, לִפְנֵי רוֹדָה אַיְיבֶּל מִדְּרוֹם אַפְרִיקָה (13.0 שְׁנִיּוֹת) וְדָלִית קְלַיְינְמָן מִיִּשְׂרָאֵל (13.2 שְׁנִיּוֹת) וּבְכָךְ הֵגֵנָּה עַל תּוֹאֳרָהּ, בְּרִיצַת 200 מֶטֶר בְּתוֹצָאָה שֶׁל 26.7 שְׁנִיּוֹת, לִפְנֵי רוֹדָה אַיְיבֶּל (27.3 שְׁנִיּוֹת) וְרוּת הֶרְשְׁלֶר מִיִּשְׂרָאֵל (27.7 שְׁנִיּוֹת) וּבִקְפִיצָה לָרוֹחַק בְּשִׂיא מַכַּבִּיָּה חָדָשׁ שֶׁל 5.36 מֶטֶר, לִפְנֵי תָּמָר מֵיטָל-שׁוּמָכֶר (4.87 מֶטֶר) וּמִרְיָם טֵפֶר (4.86 מֶטֶר), שְׁתֵּיהֶן מִיִּשְׂרָאֵל.[3]
 אַדְרִיכָלוּת קוֹנְסְטְרוּקְטִיבִיסְטִית
אַדְרִיכָלוּת קוֹנְסְטְרוּקְטִיבִיסְטִית הוּא סִגְנוֹן בָּאַדְרִיכָלוּת הַמּוֹדֶרְנִית שֶׁשִּׂגְשֵׂג בִּבְרִית הַמּוֹעָצוֹת בִּשְׁנוֹת הָ-20 וּבִשְׁנוֹת הַ-30 הַמּוּקְדָּמוֹת, כְּחֵלֶק מִתְּנוּעַת הַקּוֹנְסְטְרוּקְטִיבִיזְם. סִגְנוֹן זֶה שִׁילֵּב טֶכְנוֹלוֹגְיָה וְהַנְדָּסָה מִתְקַדְּמִים יַחַד עִם מַטָּרָה חֶבְרָתִית קוֹמוּנִיסְטִית מוּצְהֶרֶת. הַסִּגְנוֹן הִתְחַלֵּק לְמִסְפַּר זִרְמֵי מִשְׁנֶה שֶׁהִתְחָרוּ אֶחָד בַּשֵּׁנִי וְנוֹצְרוּ בּוֹ פְּרוֹיֶקְטִים עִיצּוּבִיִּים חֲלוּצִיִּים שֶׁלֹּא הִגִּיעוּ לִידֵי מִימּוּשׁ, כְּמוֹ גַּם מִבְנִים מוּגְמָרִים. תְּחִילָּתוֹ שֶׁל הַסִּגְנוֹן הָיְיתָה לַכִּיווּן הַתֵּאוֹרֵטִי וְהָאָוַונְגַּרְדִּי אוּלָם בַּמַּחֲצִית הַשְּׁנִיָּיה שֶׁל שְׁנוֹת הָ-20, כְּשֶׁנִּדְרְשׁוּ הָאַדְרִיכָלִים לִבְנוֹת בִּנְיְינֵי מְגוּרִים, בִּנְיְינֵי מִשְׂרָדִים וּמִפְעָלִים, קִיבֵּל הַסִּגְנוֹן אוֹפִי מַעֲשִׂי יוֹתֵר וְחָלַק קַוִּים מְשׁוּתָּפִים עַם הַסִּגְנוֹן הַבֵּינְלְאוּמִּי.[1]
בִּסְבִיבוֹת 1932 יָרְדָה קַרְנוֹ שֶׁל הַסִּגְנוֹן וְסִגְנוֹן הַקְּלָסִיצִיזְם הַסְּטָלִינִיסְטִי הָפַךְ לְשׁוֹלֵט בַּבְּנִיָּה בִּבְרִית הַמּוֹעָצוֹת. בְּבִנְיְינֵי מְגוּרִים שֶׁל הָאַדְרִיכָלוּת הַקּוֹנְסְטְרוּקְטִיבִיסְטִית, דּוּגְמַת הַנַּרְקוֹמְפִין (בְּמוֹסְקְבָה) נַעֲשָׂה נִיסָּיוֹן לְעַצֵּב אֶת חַיֵּי הַדַּיָּירִים בְּעֶזְרַת תִּכְנוּן הַבִּנְיָין: הָיָה בּוֹ שִׁיתּוּף בְּשֵׁירוּתִים כְּגוֹן: מִטְבָּח, כְּבִיסָה, שִׁטְחֵי הִתְכַּנְּסוּת וְחִינּוּךְ מְשׁוּתָּף שֶׁל הַיְּלָדִים וְהוּשַׂם דָּגֵשׁ עַל שִׁוְויוֹן בֵּין הַמִּינִים. כׇּל אֵלּוּ נִזְנְחוּ בִּשְׁנוֹת הַ-30 וְהוּחְלְפוּ בִּתְחִיַּית עֶרְכֵי הַמִּשְׁפָּחָה שֶׁהָיְיתָה חֵלֶק מֵהַמִּשְׁנָה הַסְּטָאלִינִיסְטִית.
תִּכְנוּן עָרִים בָּעִידָּן הַקּוֹנְסְטְרוּקְטִיבִיסְטִי
לַמְרוֹת שְׁאַפְתָּנוּתָן שֶׁל רַבּוֹת מֵהַצָּעוֹת הָאַדְרִיכָלִים הַקּוֹנְסְטְרוּקְטִיבִיסְטִיִּים לְתִכְנוּן שְׁכוּנוֹת וְעָרִים מִן הַיְּסוֹד, מְעַטּוֹת הֵן הַדּוּגְמָאוֹת לְתִכְנוּן עָרִים עִקְבִי וּמְגוּבָּשׁ שֶׁנּוֹצַר עַל יָדָם. רוֹבַע נַרְוְוסְקָיָה זַסְטָוָוה שֶׁבְּלֵנִינְגְרָד הָיָה אֶחָד מֵהַמּוֹקְדִים שֶׁל הָאַדְרִיכָלוּת הַקּוֹנְסְטְרוּקְטִיבִיסְטִית. הָחֵל מֵ-1925 נִבְנוּ בְּלֵנִינְגְרָד בָּתֵּי מְגוּרִים לְפוֹעֲלִים וְגַם מִבְנֵי צִיבּוּר. רַבִּים מֵהַקּוֹנְסְטְרוּקְטִיבִיסְטִיִּים קִיוּוּ לְכָךְ שֶׁבְּתוֹכְנִית הַחוֹמֶשׁ הָרִאשׁוֹנָה (1928) וּבַ"מַּהְפֵּכָה הַתַּרְבּוּתִית" שֶׁנִּלְוְותָה אֵלֶיהָ, תִּנָּתֵן לָהֶם אֶפְשָׁרוּת לְבַטֵּא אֶת רַעְיוֹנוֹתֵיהֶם הָאַדְרִיכָלִיִּים, אַךְ בְּאוֹתָהּ עֵת נֶחְלַק הַסִּגְנוֹן בֵּין חֲסִידֵי הַתִּכְנוּן הָאוּרְבָּנִי וּבֵין אֵלּוּ שֶׁנָּטוּ אַחֲרֵי הַתִּכְנוּן הַלֹּא אוּרְבָּנִי, שֶׁהָרַעְיוֹנוֹת הַבּוֹלְטִים בּוֹ הָיוּ עִיר הַגַּנִּים וְהָעִיר הַלִּינֵיאָרִית. נוֹסָף לְכָךְ, בְּרוֹב הַמִּקְרִים הוֹעׇדְפוּ תּוֹכְנִיּוֹת פְּרַגְמָטִיּוֹת יוֹתֵר, כְּמוֹ לְמָשָׁל, בִּבְנִיַּית הָעִיר הַחֲדָשָׁה מַגְנִיטוֹגוֹרְסְק שֶׁבְּהָרֵי אוּרָל בָּהּ הוֹעׇדְפוּ תּוֹכְנִיּוֹתֵיהֶם שֶׁל קְבוּצַת אַדְרִיכָלִים גֶּרְמָנִים, בְּרָאשׁוּת אֵרְנְסְט מַאי.
 אֵאוּדוֹקְסוֹס מַקְנִידוֹס
אֵאוּדוֹקְסוֹס מַקְנִידוּס (בִּיוָונִית: Εύδοξος, 408 אוֹ 410 - 347 אוֹ 355 לפנה"ס) הָיָה אַסְטְרוֹנוֹם, מָתֵמָטִיקַאי, פִיזִיקַאי וּמְלוּמָּד יְוָונִי, מִתַּלְמִידֵי אַפְּלָטוֹן. לְאוֹר הָעוּבְדָּה שֶׁכׇּל חִיבּוּרָיו אָבְדוּ, יְדִיעוֹתֵינוּ אוֹדוֹתָיו מוּשָּׂגוֹת הוֹדוֹת לִמְקוֹרוֹת מִשְׁנִיִּים, דּוּגְמַת הַפּוֹאֵמָה שֶׁל אָרָאטוֹס עַל אַסְטְרוֹנוֹמְיָה. הַסְּפָרִיקָה שֶׁל תֵּאוֹדוֹסְיוֹס מִבִּיתִינְיָה עֲשׂוּיָה לִהְיוֹת מְבוּסֶּסֶת עַל חִיבּוּר שֶׁל אֵאוּדוֹקְסוֹס.
אֵאוּדוֹקְסוֹס הָיָה בְּנוֹ שֶׁל אָאֶסְכִּינֶס מַקְנִידוֹס שֶׁבְּאַסְיָה הַקְּטַנָּה. מַסָּעוֹ הָרִאשׁוֹן שֶׁל אֵאוּדוֹקְסוֹס הָיָה לְטָארֶנְטוּם עַל מְנַת לִלְמוֹד יַחַד עִם אַרְכִיטָס, מִמֶּנּוּ לָמַד מָתֵמָטִיקָה. בְּעוֹדוֹ שׁוֹהֶה בְּאִיטַלְיָה, בִּיקֵּר אֵאוּדוֹקְסוֹס בְּסִיצִילְיָה, וּבָהּ לָמַד רְפוּאָה עִם פִּילִיסְטוֹן.
בִּשְׁנַת 387 לפנה"ס לְעֵרֶךְ, בְּעוֹדוֹ בֶּן 23 שָׁנִים, נָסַע אֵאוּדוֹקְסוֹס עִם הָרוֹפֵא תֵּאוֹמְדוֹן לְאָתוּנָה כְּדֵי לִלְמוֹד יַחַד עִם מַמְשִׁיכֵי דַּרְכּוֹ שֶׁל סוֹקְרָטֶס. לְבַסּוֹף, הוּא הָיָה לְתַלְמִידוֹ שֶׁל אַפְּלָטוֹן, עִמּוֹ לָמַד בְּמֶשֶׁךְ חוֹדָשִׁים מִסְפָּר, אַךְ בְּשֶׁל חִילּוּקֵי-דֵּעוֹת נִפְרְדוּ דַּרְכֵיהֶם. אֵאוּדוֹקְסוֹס הָיָה עָנִי לְמַדַּי וְיָכוֹל הָיָה לְהַרְשׁוֹת לְעַצְמוֹ לְהִתְגּוֹרֵר בְּפִּירֵאוּס בִּלְבַד. עַל מְנַת לְהַגִּיעַ לְשִׁיעוּרָיו שֶׁל אַפְּלָטוֹן, הוּא נֶאֱלַץ לִצְעוֹד שִׁבְעָה מִילִין לְכׇל כִּיווּן, בְּכׇל יוֹם. בְּשֶׁל עוֹנְיוֹ, חֲבֵרָיו גִּייְּסוּ תְּרוּמוֹת שֶׁהִסְפִּיקוּ כְּדֵי לְשׇׁלְחוֹ לְהֶלְיוֹפּוֹלִיס בְּמִצְרַיִם כְּדֵי שֶׁיַּחְקוֹר שָׁם אֶת תְּחוּמֵי הָאַסְטְרוֹנוֹמְיָה וְהַמָּתֵמָטִיקָה, וְשָׁם הוּא הִתְגּוֹרֵר בְּמֶשֶׁךְ 16 חוֹדָשִׁים. מִמִּצְרַיִם, הוּא נָסַע לְקִיזִיקוּס, הַשּׁוֹכֶנֶת לְחוֹפוֹ הַדְּרוֹמִי שֶׁל יַם הַשַּׁיִשׁ, וּלְאַחַר מִכֵּן נָשְׂאוּ אוֹתוֹ רַגְלָיו לִפְּרוֹפּוֹנְטִיס. לְאַחַר מִכֵּן, הִמְשִׁיךְ אֵאוּדוֹקְסוֹס דָּרוֹמָה וְהִגִּיעַ לַחֲצַר מָאוּסוֹלוֹס. בְּמַהֲלַךְ מַסְּעוֹתָיו קִיבֵּץ לוֹ אֵאוּדוֹקְסוֹס תַּלְמִידִים רַבִּים מִמְּחוֹזוֹת שׁוֹנִים.
בִּתְחוּם הָאַסְטְרוֹנוֹמְיָה הוּא הֵחֵל לְהִשְׁתַּמֵּשׁ בַּמּוּנָּח גְּלוֹבּוּס וְתָרַם לְחֵקֶר תְּנוּעַת כּוֹכְבֵי לֶכֶת. בִּתְחוּם הַמָּתֵמָטִיקָה הוּא חָקַר יַחַס בֵּין גְּדָלִים וּבִיסֵּס אֶת שִׁיטַת הַמִּיצּוּי. עַל שְׁמוֹ קְרוּיִים מַכְתְּשִׁים בְּמַאֲדִים וּבַיָּרֵחַ.
 אוּזְבֵּקִיסְטָן בַּמִּשְׂחָקִים הָאוֹלִימְפִּיִּים
אוּזְבֵּקִיסְטָן מִשְׁתַּתֶּפֶת בַּמִּשְׂחָקִים הָאוֹלִימְפִּיִּים מֵאָז אוֹלִימְפִּיָּאדַת אַטְלַנְטָה (1996). עַד 1991 הָיְיתָה אוּזְבֵּקִיסְטָן חֵלֶק מִבְּרִית הַמּוֹעָצוֹת, וּסְפּוֹרְטָאֶיהָ הִתְחָרוּ בְּמִסְגֶּרֶת הַמִּשְׁלַחַת הַסּוֹבְיֵיטִית. בְּאוֹתָהּ שָׁנָה הִתְפָּרְקָה בְּרִית הַמּוֹעָצוֹת לַ-15 מְדִינוֹת נִפְרָדוֹת, וּ-12 מֵהֶן, בָּהֶן אוּזְבֵּקִיסְטָן, שָׁלְחוּ מִשְׁלַחַת מְאוּחֶדֶת לְאוֹלִימְפִּיָּאדַת בַּרְצֵלוֹנָה (1992). בְּאַרְבַּע הָאוֹלִימְפִּיָּאדוֹת בָּהֶן הִשְׁתַּתְּפָה אוּזְבֵּקִיסְטָן, צָבְרוּ הַסְּפּוֹרְטָאִים הָאוּזְבֵּקִים 32 מֵדַלְיוֹת, מֵהֵן 9 מֵדַלְיוֹת זָהָב. הָעֲנָפִים הַבּוֹלְטִים בְּיוֹתֵר הֵם הֵיאָבְקוּת וְאִיגְרוּף.
הִיסְטוֹרְיָה
הַמִּשְׁלַחַת הָעַצְמָאִית הָרִאשׁוֹנָה שֶׁל אוּזְבֵּקִיסְטָן, בְּאוֹלִימְפִּיָּאדַת אַטְלַנְטָה, כָּלְלָה 71 סְפּוֹרְטָאִים. שְׁנַיִים מֵהֶם זָכוּ בַּמֵּדַלְיוֹת הָרִאשׁוֹנוֹת בְּתוֹלְדוֹתֶיהָ: הַגּ'וּדוֹקָא אַרְמָן בַּגְדֵסָרוֹב זָכָה בְּמֵדַלְיַית כֶּסֶף וְהַמִּתְאַגְרֵף כָּרִים טוֹלְגָנוֹב זָכָה בְּמֵדַלְיַית אָרָד. בְּאוֹלִימְפִּיָּאדַת סִידְנִי (2000) נָטְלוּ חֵלֶק 70 סְפּוֹרְטָאִים אוּזְבֵּקִים. הַמִּתְאַגְרֵף מַחְמַטְקוֹדִיר עַבְּדוּלָאיֶיב זָכָה בְּמֵדַלְיַית הַזָּהָב הָרִאשׁוֹנָה בְּתוֹלְדוֹת אוּזְבֵּקִיסְטָן (בְּמִשְׂחֲקֵי הַקַּיִץ). שְׁנֵי מִתְאַגְרְפִים נוֹסָפִים, סֶרְגֵּיי מִיחָאִילוֹב וְרוֹסְטָם סָאִידוֹב, זָכוּ בְּמֵדַלְיוֹת אָרָד, וּמִתְאַגְרֵף רְבִיעִי, רוֹסְלָאן צָ'גָאיֶיב, הֶעְפִּיל לִשְׁלַב רֶבַע גְּמָר. בְּהֵיאָבְקוּת בָּלַט אַרְתּוּר טָאִימְזוֹב, שֶׁזָּכָה בְּמֵדַלְיַית כֶּסֶף.
גַּם בְּאוֹלִימְפִּיָּאדַת אָתוּנָה (2004) הִשְׁתַּתְּפוּ 70 סְפּוֹרְטָאִים אוּזְבֵּקִים. טָאִימְזוֹב זָכָה בְּמֵדַלְיַית זָהָב, וְכָמוֹהוּ גַּם הַמִּתְאַבֵּק אָלֶכְּסַנְדֶר דוֹקטוֹרְשְׁוִוילִי. מִתְאַבֵּק שְׁלִישִׁי, מגומד אִיבְּרָגִימוֹב זָכָה בְּמֵדַלְיַית כֶּסֶף. וְהַמִּתְאַגְרְפִים בָּחוּדִירְג'וֹן סוֹלְטוֹנוֹב וְאוֹטְקִירְבֶּק חַיְידָרוֹב זָכוּ בְּמֵדַלְיוֹת אָרָד, וּשְׁלוֹשָׁה מִתְאַגְרְפִים אוּזְבֵּקִים נוֹסָפִים, טוֹלַאשְׁבּוֹי דּוּנְיוּרוֹב, דִילְשׁוֹד מַחְמוּדוֹב וְשֶׁרְזוּד חוּסָאנוֹב, הֶעְפִּילוּ לִשְׁלַב רֶבַע הַגְּמָר. בַּהֲרָמַת מִשְׁקוֹלוֹת בָּלַט אָלֶכְּסַנְדֶר אוּרִינוֹב, שֶׁסִּייֵּם בַּמָּקוֹם הַשְּׁמִינִי. עַל הִשְׁתַּתְּפוּתָהּ שֶׁל אוּזְבֵּקִיסְטָן בְּמִשְׂחָקִים אֵלֶּה הֵעִיבָה פְּסִילָתָהּ שֶׁל מְיַידָּה הַפַּטִּישׁ אוֹלְגָה שְׁצ'וּקִינָה, בְּשֶׁל שִׁימּוּשׁ בְּסַמִּים.
לְאוֹלִימְפִּיָּאדַת בֶּיְיגִ'ינְג (2008) שָׁלְחָה אוּזְבֵּקִיסְטָן מִשְׁלַחַת קְטַנָּה יוֹתֵר, בַּת 58 סְפּוֹרְטָאִים. הַמִּתְאַבֵּק אַרְתּוּר טָאִימְזוֹב זָכָה בְּמֵדַלְיַית זָהָב שְׁנִיָּיה בִּרְצִיפוּת, וַעֲמִיתוֹ סוֹסְלָן טִיגָיֶיב זָכָה בְּמֵדַלְיַית כֶּסֶף, אַךְ זוֹ נִשְׁלְלָה מִמֶּנּוּ בְּ-2016 בְּשֶׁל הַרְשָׁעָה בְּאִמְרוּץ. דִילְשׁוֹד מַנְסוּרוֹב סִייֵּם בַּמָּקוֹם הַחֲמִישִׁי וְקוּרְבָּן קוּרְבָּנוֹב בַּמָּקוֹם הַשְּׁבִיעִי. הַגּ'וּדוֹקָא עַבְּדוּלוּ טַנְגָרָיֶיב זָכָה בְּמֵדַלְיַית כֶּסֶף, בְּעוֹד שֶׁרִישׁוּד סובִּירוֹב זָכָה בְּמֵדַלְיַית אָרָד. ג'וּדוֹקָא נוֹסָף, מִירָאלִי שְׁרִיפּוֹב, סִייֵּם בַּמָּקוֹם הַחֲמִישִׁי. בְּנוֹסָף, שְׁנֵי מִתְעַמְּלִים אוּזְבֵּקִים זָכוּ לָרִאשׁוֹנָה בְּמֵדַלְיוֹת אָרָד - אַנְטוֹן פּוּקִין בְּתַרְגִּיל הַמַּקְבִּילִים וְיַקְטָרִינָה חִילְקוֹ, שֶׁהָיְיתָה לַסְּפּוֹרְטָאִית הָאוּזְבֵּקִית הָרִאשׁוֹנָה, וְהַיְּחִידָה עַד כֹּה, שֶׁזּוֹכָה בַּמֵּדַלְיָה הָאוֹלִימְפִּית בְּמִשְׂחֲקֵי הַקַּיִץ - בִּטְרַמְפּוֹלִינָה. חוֹתֵר הַקָּאנוּ וָאדִים מֶנְקוֹב הָיָה קָרוֹב לִזְכִיָּיה רִאשׁוֹנָה בְּמֵדַלְיָה גַּם בְּעָנָף זֶה, אַךְ סִייֵּם בַּמָּקוֹם הָרְבִיעִי בְּמִקְצֵה הַסְּפְּרִינְט לְ-1,000 מֶטֶר. בַּעֲנַף הַטַּאקְוַונְדּוֹ בָּלַט אָקָמָאל אִירַגְשֵׁב, שֶׁסִּייֵּם בַּמָּקוֹם הַחֲמִישִׁי. הַמִּתְאַגְרְפִים הֶאוּזְבֵּקִים לֹא הִצְלִיחוּ בְּמִשְׂחָקִים אֵלּוּ לִזְכּוֹת בְּמֵדַלְיוֹת, אַךְ שְׁנַיִים מֵהֶם, דִילְשׁוֹד מַחְמוּדוֹב וְאַלְשׁוּד רָסוּלוֹב, הֶעְפִּילוּ לִשְׁלַב רֶבַע הַגְּמָר.
בְּאוֹלִימְפִּיָּאדַת לוֹנְדוֹן (2012) הִשְׁתַּתְּפוּ 54 סְפּוֹרְטָאִים אוּזְבֵּקִים. טָאִימְזוֹב זָכָה בְּמֵדַלְיַית זָהָב שְׁלִישִׁית בִּרְצִיפוּת, שֶׁהָיְיתָה גַּם מֵדַלְיַית הַזָּהָב הַיְּחִידָה לִזְכוּתָהּ שֶׁל הַמִּשְׁלַחַת. מִתְאַבֵּק נוֹסָף, סוּסְלָאן טִיגָיֶיב, זָכָה בְּמֵדַלְיַית אָרָד, אַךְ זוֹ נִשְׁלְלָה מִמֶּנּוּ חוֹדָשִׁים סְפוּרִים לְאַחַר הַמִּשְׂחָקִים, לְאַחַר שֶׁנִּמְצָא כִּי נָטַל סַמִּים. שְׁנֵי סְפּוֹרְטָאִים אוּזְבֵּקִים נוֹסָפִים שֶׁזָּכוּ בַּמֵּדַלְיוֹת הָיוּ הַגּ'וּדוֹקָא רִישׁוּד סובִּירוֹב וְהַמִּתְאַגְרֵף אַבּוּס אָטוֹיֶב, שֶׁזָּכוּ בְּמֵדַלְיוֹת אָרָד. אַרְבָּעָה מִתְאַגְרְפִים נוֹסָפִים הוּדְּחוּ בִּשְׁלַב רֶבַע הַגְּמָר. סְפּוֹרְטָאִים אוּזְבֵּקִים נוֹסָפִים שֶׁבָּלְטוּ בְּלוֹנְדוֹן כָּלְלוּ אֶת חוֹתֵר הַקָּאנוּ וָאדִים מֶנְקוֹב (מָקוֹם רְבִיעִי בַּמִּקְצֶה לְ-1000 מֶטְרִים), מֵרִים הַמִּשְׁקוֹלוֹת רוֹסְלָן נוֹרוֹדִינוֹב (מָקוֹם הָרְבִיעִי), רוֹכֵב הָאוֹפַנַּיִים סֶרְגֵיי לָגוֹטִין (מָקוֹם חֲמִישִׁי בְּמֵרוֹץ אוֹפַנֵּי כְּבִישׁ), הַגּ'וּדוֹקָא רַמְזִידִין (מָקוֹם חֲמִישִׁי), הַמִּתְאַבֵּק קוּרְבָּן קוּרְבָּנוֹב (מָקוֹם חֲמִישִׁי), מֵרִים הַמִּשְׁקוֹלוֹת אִיוָואן אַפְרָמוֹב (מָקוֹם חֲמִישִׁי); הָאַתְלֵטִית סְבֶטְלָנָה רַדְזִיוִויל (מָקוֹם שְׁבִיעִי בִּקְפִיצָה לַגּוֹבַהּ) וְהַגּ'וּדוֹקָא דִילְשׁוֹד צ'וּרָיֶיב (מָקוֹם שְׁבִיעִי).
אוֹלִימְפִּיָּאדַת רִיּוֹ דֶה זָ'נֵיירוֹ (2016) הָיְיתָה הַמּוּצְלַחַת בְּיוֹתֵר בְּתוֹלְדוֹת אוּזְבֵּקִיסְטָן, עִם 13 מֵדַלְיוֹת, מֵהֵן 4 מֵדַלְיוֹת זָהָב, הֶישֵּׂג דּוֹמֶה לְהֶישֵּׂגֵי הַמִּשְׁלָחוֹת הָאוּזְבֵּקִיּוֹת בִּשְׁלוֹשׁ הָאוֹלִימְפִּיָּאדוֹת הַקּוֹדְמוֹת גַּם יַחַד. הָאִיגְרוּף הָיָה הֶעָנָף הַבּוֹלֵט בְּיוֹתֵר. הַמִּתְאַגְרְפִים הִתְחָרוּ בְּכׇל עֶשֶׂר קָטֵגוֹרְיוֹת הַמִּשְׁקָל לִגְבָרִים, וּבְשֶׁבַע מֵהֶן זָכוּ בְּמֵדַלְיוֹת. חָסַנְבּוֹי דּוּסְמָטוֹב, שָׁאחוֹבִּידִין זוֹיְרוֹב וּפַּזְלִידִין גַאיְבַּנְזָארוֹב זָכוּ בְּמֵדַלְיוֹת זָהָב, שַׁחְרָאם גִיאָסוֹב וּבַקְטָמִיר מָלִיקוֹזָיֶיב זָכוּ בְּמֵדַלְיוֹת כֶּסֶף, וּמוֹרוֹג'וֹן אַחְמַדְלָיֶיב וְרוֹסְטָם טוֹלְגָנוֹב זָכוּ בְּמֵדַלְיוֹת אָרָד. בְּנוֹסָף, סִייְּמוּ הוֹרְשִׁיד טוֹזִ'יבָּיֶיב וּבָחוֹדִיר זֵ'לוֹלוֹב בַּמָּקוֹם הַחֲמִישִׁי. בְּמֵדַלְיַית הַזָּהָב הָרְבִיעִית לִזְכוּת הָאוּזְבֵּקִים זָכָה מֵרִים הַמִּשְׁקוֹלוֹת רוֹסְלָאן נוֹרוֹדִינוֹב. אִיוָואן אַפְרָמוֹב סִייֵּם בַּמָּקוֹם הַחֲמִישִׁי וְרוֹסְטָם גַ'נְגַבָּיֶיב בַּמָּקוֹם הַשִּׁישִּׁי. בְּהֵיאָבְקוּת זָכוּ שְׁלוֹשָׁה אוּזְבֵּקִים בְּמֵדַלְיוֹת אָרָד: אִיחַטְיוֹר נַבְּרוֹזוֹב, מֵגוּמָד אִיבְּרָגִימוֹב וְאַלְמוֹרָט טַסְמוֹרָדוֹב. בַּקְזוּד עַבּדוּרַחְמוֹנוֹב סִייֵּם בַּמָּקוֹם הַחֲמִישִׁי. הַגּ'וּדוֹקָא רִישׁוּד סוֹבִּירוֹב זָכָה בְּמֵדַלְיַית אָרָד שְׁלִישִׁית בִּרְצִיפוּת, וְגַם דיוּרְבֶּק אוֹרוֹזְבּוֹיֶיב זָכָה בְּמֵדַלְיַית אָרָד. עַבְּדוּלוּ טַנְגָרָיֶיב סִייֵּם בַּמָּקוֹם הַחֲמִישִׁי. סְפּוֹרְטָאִים אוּזְבֵּקִים נוֹסָפִים שֶׁבָּלְטוּ בְּרִיּוֹ דֶה זָ'נֵיירוֹ כָּלְלוּ אֶת לוֹחֵם הַטַּאקְוַונְדּוֹ דֵמִיטְרִי שׁוֹקִין שֶׁסִּייֵּם בַּמָּקוֹם הַחֲמִישִׁי, אֶת חוֹתֵר הַקָּאנוּ גְרָאסִים קוּצְ'נֶב שֶׁסִּייֵּם בַּמָּקוֹם הַשְּׁבִיעִי בַּמִּקְצֶה לְ-1000 מֶטֶר וּבַמָּקוֹם הַשְּׁמִינִי בְּמִקְצֵה הַזּוּגוֹת לְ-1000 מֶטֶר, יַחַד עִם סָרִיק מִירְבֵּקוֹב, וְאֶת הַמִּתְעַמֶּלֶת הַוָּותִיקָה אוֹקְסָנָה צ'וסוביטינה בַּת הָ-41, שֶׁסִּייְּמָה בַּמָּקוֹם הַשְּׁבִיעִי בְּתַרְגִּיל הַקְּפִיצוֹת.
מִשְׂחֲקֵי הַחוֹרֶף
אוֹלִימְפִּיָּאדַת הַחוֹרֶף הָרִאשׁוֹנָה בָּהּ הִשְׁתַּתְּפָה אוּזְבֵּקִיסְטָן הָיְיתָה אוֹלִימְפִּיָּאדַת לִילֵהָאמֶר (1994). הָיְיתָה זוֹ אוֹלִימְפִּיָּאדַת הַחוֹרֶף הַיְּחִידָה עַד כֹּה בָּהּ זָכָה סְפּוֹרְטַאי אוּזְבֵּקִי בְּמֵדַלְיָה - לִינָה צַ'רְיָאזוֹבָה זָכְתָה בְּמֵדַלְיַית זָהָב בִּסְקִי חוֹפְשִׁי. פְּרָט לְהֶישֵּׂג זֶה, הַהֶישֵּׂג הַגָּבוֹהַּ הַיָּחִיד הַנּוֹסָף שֶׁל הַמִּשְׁלַחַת הָאוּזְבֵּקִית נִרְשַׁם בְּאוֹלִימְפִּיָּאדַת נָאגָנוֹ (1998), אָז דּוֹרְגָה הַמַּחְלִיקָה הָאוֹמָּנוּתִית טַטְיָאנָה מֵלִינִינָה בַּמָּקוֹם הַשְּׁמִינִי.

 אִיבְּרָאהִים סוּס
אִיבְּרָאהִים סוּס (נוֹלַד בִּשְׁנַת 1945 בִּירוּשָׁלַיִם), הוּא סוֹפֵר וּפּוֹלִיטִיקַאי פָלַסְטִינִי.
בִּיּוֹגְרַפְיָה
לָמַד בְּאוּנִיבֶרְסִיטָאוֹת בְּפָרִיז וּבְלוֹנְדוֹן, אֶת הַדּוֹקְטוֹרָאט הוּא עָשָׂה בְּמַדְּעֵי הַמְּדִינָה בְּפָרִיז.
נוֹסָף עַל הַקַּרְיֵירָה הַפּוֹלִיטִית שֶׁלּוֹ, הוּא גַּם פְּסַנְתְּרָן, מַלְחִין וְסוֹפֵר. בַּשָּׁנִים 1978 – 1992 הוּא יִיצֵּג אֶת אָשָׁ"ף בְּפָרִיז[1]
, בַּשָּׁנִים 1975 – 1980 בְּאוּנֶסְקוֹ וּבַשָּׁנִים 1983 – 1985 בְּסֵנֵגָל. בַּשָּׁנִים 1998 – 2003 הוּא כִּיהֵן בִּסְגַן מְנַהֵל אִרְגּוּן הָעֲבוֹדָה הַבֵּינְלְאוּמִּי, וְהָחֵל מִשְּׁנַת 2003 סְגַן נְשִׂיא אוּנִיבֶרְסִיטַת אַל-קוּדְס בִּירוּשָׁלַיִם.
